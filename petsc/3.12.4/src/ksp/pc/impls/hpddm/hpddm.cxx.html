127:   while (i < PETSC_HPDDM_MAXLEVELS) {
128:     PetscInt p = 1;

131:       PetscNewLog(pc, data->levels + i - 1);
132:     }

134:     /* if the previous level has a single process, it is not possible to coarsen further */

177: /*
178:      PCHPDDMGetComplexities - Computes the grid and operator complexities.

180:    Input Parameter:
181: .     pc - preconditioner context

183:    Output Parameters:
184: +     gc - grid complexity = sum_i(m_i) / m_1
185: .     oc - operator complexity = sum_i(nnz_i) / nnz_1

187:    Notes:
188:      PCGAMG does not follow the usual convention and names the grid complexity what is usually referred to as the operator complexity. PCHPDDM follows what is found in the literature, and in particular, what you get with PCHYPRE and -pc_hypre_boomeramg_print_statistics.

190:    Level: advanced

192: .seealso:  PCMGGetGridComplexity(), PCHPDDM
193: */
194: static PetscErrorCode PCHPDDMGetComplexities(PC pc, PetscReal *gc, PetscReal *oc)
195: {
196:   PC_HPDDM       *data = (PC_HPDDM*)pc->data;
197:   MatInfo        info;
198:   PetscInt       n, m, nnz1 = 1, m1 = 1;

202:   for (n = 0, *gc = 0, *oc = 0; n < data->N; ++n) {
203:     if (data->levels[n]->ksp) {
204:       Mat P;
205:       KSPGetOperators(data->levels[n]->ksp, NULL, &P);
206:       MatGetSize(P, &m, NULL);
207:       *gc += m;
208:       MatGetInfo(P, MAT_GLOBAL_SUM, &info);
209:       *oc += (PetscReal)info.nz_used;
210:       if (n == 0) {
211:         m1 = m;
212:         nnz1 = info.nz_used;
213:       }
214:     }
215:   }
216:   *gc /= (PetscReal)m1;
217:   *oc /= (PetscReal)nnz1;
218:   return(0);
219: }

311: PETSC_STATIC_INLINE PetscErrorCode PCHPDDMDeflate_Private(PC pc, Vec x, Vec y)
312: {
313:   PC_HPDDM_Level *ctx;
314:   PetscScalar    *out;

318:   PCShellGetContext(pc, (void**)&ctx);
319:   /* going from PETSc to HPDDM numbering */
320:   VecScatterBegin(ctx->scatter, x, ctx->v[0][0], INSERT_VALUES, SCATTER_FORWARD);
321:   VecScatterEnd(ctx->scatter, x, ctx->v[0][0], INSERT_VALUES, SCATTER_FORWARD);
322:   VecGetArray(ctx->v[0][0], &out);
323:   ctx->P->deflation<false>(NULL, out, 1); /* y = Q x */
324:   VecRestoreArray(ctx->v[0][0], &out);
325:   /* going from HPDDM to PETSc numbering */
326:   VecScatterBegin(ctx->scatter, ctx->v[0][0], y, INSERT_VALUES, SCATTER_REVERSE);
327:   VecScatterEnd(ctx->scatter, ctx->v[0][0], y, INSERT_VALUES, SCATTER_REVERSE);
328:   return(0);
329: }

331: /*MC
332:      PCHPDDMShellApply - Applies a (2) deflated, (1) additive, or (3) balanced coarse correction. In what follows, E = Z Pmat Z^T and Q = Z^T E^-1 Z.

334: .vb
335:    (1) y =                Pmat^-1              x + Q x,
336:    (2) y =                Pmat^-1 (I - Amat Q) x + Q x (default),
337:    (3) y = (I - Q Amat^T) Pmat^-1 (I - Amat Q) x + Q x.
338: .ve

340:    Input Parameters:
341: +     pc - preconditioner context
342: .     x - input vector
343: -     y - output vector

345:    Application Interface Routine: PCApply()

347:    Notes:
348:      The options of Pmat^1 = pc(Pmat) are prefixed by -pc_hpddm_levels_1_pc_. Z is a tall-and-skiny matrix assembled by HPDDM. The number of processes on which (Z Pmat Z^T) is aggregated is set via -pc_hpddm_coarse_p.
349:      The options of (Z Pmat Z^T)^-1 = ksp(Z Pmat Z^T) are prefixed by -pc_hpddm_coarse_ (KSPPREONLY and PCCHOLESKY by default), unless a multilevel correction is turned on, in which case, this function is called recursively at each level except the coarsest one.
350:      (1) and (2) visit the "next" level (in terms of coarsening) once per application, while (3) visits it twice, so it is asymptotically twice costlier. (2) is not symmetric even if both Amat and Pmat are symmetric.

352: .seealso:  PCHPDDM, PCHPDDMCoarseCorrectionType
353: M*/
354: static PetscErrorCode PCHPDDMShellApply(PC pc, Vec x, Vec y)
355: {
356:   PC_HPDDM_Level *ctx;
357:   Mat            A;

361:   PCShellGetContext(pc, (void**)&ctx);
362:   if (ctx->P) {
363:     KSPGetOperators(ctx->ksp, &A, NULL);
364:     PCHPDDMDeflate_Private(pc, x, y);                    /* y = Q x                          */
365:     if (ctx->parent->correction == PC_HPDDM_COARSE_CORRECTION_DEFLATED || ctx->parent->correction == PC_HPDDM_COARSE_CORRECTION_BALANCED) {
366:       MatMult(A, y, ctx->v[1][0]);                       /* y = A Q x                        */
367:       VecWAXPY(ctx->v[1][1], -1.0, ctx->v[1][0], x);     /* y = (I - A Q) x                  */
368:       PCApply(ctx->pc, ctx->v[1][1], ctx->v[1][0]);      /* y = M^-1 (I - A Q) x             */
369:       if (ctx->parent->correction == PC_HPDDM_COARSE_CORRECTION_BALANCED) {
370:         MatMultTranspose(A, ctx->v[1][0], ctx->v[1][1]); /* z = A^T M^-1 (I - A Q) x         */
371:         PCHPDDMDeflate_Private(pc, ctx->v[1][1], ctx->v[1][1]);
372:         VecAXPY(ctx->v[1][0], -1.0, ctx->v[1][1]);       /* y = (I - Q A^T) M^-1 (I - A Q) x */
373:       }
374:       VecAXPY(y, 1.0, ctx->v[1][0]);                     /* y = y + Q x                      */
375:     } else if (ctx->parent->correction == PC_HPDDM_COARSE_CORRECTION_ADDITIVE) {
376:       PCApply(ctx->pc, x, ctx->v[1][0]);
377:       VecAXPY(y, 1.0, ctx->v[1][0]);                     /* y = M^-1 x + Q x                 */
378:     } else SETERRQ1(PETSC_COMM_SELF, PETSC_ERR_PLIB, "PCSHELL from PCHPDDM called with an unknown PCHPDDMCoarseCorrectionType %d", ctx->parent->correction);
379:   } else SETERRQ(PETSC_COMM_SELF, PETSC_ERR_PLIB, "PCSHELL from PCHPDDM called with no HPDDM object");
380:   return(0);
381: }

419: static PetscErrorCode PCSetUp_HPDDM(PC pc)
420: {
421:   PC_HPDDM                 *data = (PC_HPDDM*)pc->data;
422:   PC                       inner;
423:   Mat                      *sub, A, P, N, C, uaux = NULL;
424:   Vec                      xin, v;
425:   std::vector<Vec>         initial;
426:   IS                       is[1], loc, uis = data->is;
427:   ISLocalToGlobalMapping   l2g;
428:   MPI_Comm                 comm;
429:   char                     prefix[256];
430:   const char               *pcpre;
431:   const PetscScalar* const *ev;
432:   PetscInt                 n, requested = data->N;
433:   PetscBool                subdomains = PETSC_FALSE, outer = PETSC_FALSE, ismatis;
434:   DM                       dm;
435:   PetscErrorCode           ierr;

438:   if (!data->levels[0]) SETERRQ(PETSC_COMM_SELF, PETSC_ERR_PLIB, "Not a single level allocated");
439:   PCGetOptionsPrefix(pc, &pcpre);
440:   PCGetOperators(pc, &A, &P);
441:   PetscObjectGetComm((PetscObject)pc, &comm);
442:   if (!data->levels[0]->ksp) {
443:     KSPCreate(comm, &data->levels[0]->ksp);
444:     PetscSNPrintf(prefix, sizeof(prefix), "%spc_hpddm_%s_", pcpre ? pcpre : "", data->N > 1 ? "levels_1" : "coarse");
445:     KSPSetOptionsPrefix(data->levels[0]->ksp, prefix);
446:     KSPSetType(data->levels[0]->ksp, KSPPREONLY);
447:   } else {
448:     const int *addr = data->levels[0]->P ? data->levels[0]->P->getAddrLocal() : &HPDDM::i__0;

450:     if (addr != &HPDDM::i__0) {
451:       /* reuse previously computed eigenvectors */
452:       ev = data->levels[0]->P->getVectors();
453:       if (ev) {
454:         initial.reserve(*addr);
455:         VecCreateSeqWithArray(PETSC_COMM_SELF, 1, data->levels[0]->P->getDof(), ev[0], &xin);
456:         for (n = 0; n < *addr; ++n) {
457:           VecDuplicate(xin, &v);
458:           VecPlaceArray(xin, ev[n]);
459:           VecCopy(xin, v);
460:           initial.emplace_back(v);
461:           VecResetArray(xin);
462:         }
463:         VecDestroy(&xin);
464:       }
465:     }
466:     /* reset coarser levels */
467:     for (n = 1; n < PETSC_HPDDM_MAXLEVELS && data->levels[n]; ++n) {
468:       KSPDestroy(&data->levels[n]->ksp);
469:       PCDestroy(&data->levels[n]->pc);
470:     }
471:   }
472:   KSPSetOperators(data->levels[0]->ksp, A, P);

474:   PetscObjectTypeCompare((PetscObject)P, MATIS, &ismatis);
475:   if (!data->is && !ismatis) {
476:     PetscErrorCode  (*create)(DM, IS*, Mat*, PetscErrorCode (**)(Mat, PetscReal, Vec, Vec, PetscReal, IS, void*), void**) = NULL;
477:     PetscErrorCode  (*usetup)(Mat, PetscReal, Vec, Vec, PetscReal, IS, void*) = NULL;
478:     void            *uctx = NULL;

480:     /* first see if we can get the data from the DM */
481:     MatGetDM(P, &dm);
482:     if (!dm) {
483:       MatGetDM(A, &dm);
484:     }
485:     if (!dm) {
486:       PCGetDM(pc, &dm);
487:     }
488:     if (dm) { /* this is the hook for DMPLEX and DMDA */
489:       PetscObjectQueryFunction((PetscObject)dm, "DMCreateNeumannOverlap_C", &create);
490:       if (create) {
491:         (*create)(dm, &uis, &uaux, &usetup, &uctx);
492:       }
493:     }
494:     if (!create) {
495:       if (!uis) {
496:         PetscObjectQuery((PetscObject)pc, "_PCHPDDM_Neumann_IS", (PetscObject*)&uis);
497:         PetscObjectReference((PetscObject)uis);
498:       }
499:       if (!uaux) {
500:         PetscObjectQuery((PetscObject)pc, "_PCHPDDM_Neumann_Mat", (PetscObject*)&uaux);
501:         PetscObjectReference((PetscObject)uaux);
502:       }
503:     }
504:     PCHPDDMSetAuxiliaryMat(pc, uis, uaux, usetup, uctx);
505:     MatDestroy(&uaux);
506:     ISDestroy(&uis);
507:   }

509:   if (!ismatis) {
510:     PCHPDDMSetUpNeumannOverlap_Private(pc);
511:   }

513:   if (data->is || (ismatis && data->N > 1)) {
514:     if (ismatis) {
515:       std::initializer_list<std::string> list = { MATSEQBAIJ, MATSEQSBAIJ };
516:       MatISGetLocalMat(P, &N);
517:       std::initializer_list<std::string>::const_iterator it = std::find(list.begin(), list.end(), ((PetscObject)N)->type_name);
518:       MatISRestoreLocalMat(P, &N);
519:       switch (std::distance(list.begin(), it)) {
520:       case 0:
521:         MatConvert(P, MATMPIBAIJ, MAT_INITIAL_MATRIX, &C);
522:         break;
523:       case 1:
524:         /* MatCreateSubMatrices does not work with [Seq|MPI]SBAIJ and unsorted ISes, so convert to MPIBAIJ */
525:         MatConvert(P, MATMPIBAIJ, MAT_INITIAL_MATRIX, &C);
526:         MatSetOption(C, MAT_SYMMETRIC, PETSC_TRUE);
527:         break;
528:       default:
529:         MatConvert(P, MATMPIAIJ, MAT_INITIAL_MATRIX, &C);
530:       }
531:       MatGetLocalToGlobalMapping(P, &l2g, NULL);
532:       PetscObjectReference((PetscObject)P);
533:       KSPSetOperators(data->levels[0]->ksp, A, C);
534:       std::swap(C, P);
535:       ISLocalToGlobalMappingGetSize(l2g, &n);
536:       ISCreateStride(PETSC_COMM_SELF, n, 0, 1, &loc);
537:       ISLocalToGlobalMappingApplyIS(l2g, loc, &is[0]);
538:       ISDestroy(&loc);
539:     } else {
540:       is[0] = data->is;
541:       PetscOptionsGetBool(NULL, pcpre, "-pc_hpddm_define_subdomains", &subdomains, NULL);
542:       ISCreateStride(PetscObjectComm((PetscObject)data->is), P->rmap->n, P->rmap->rstart, 1, &loc);
543:     }
544:     if (data->N > 1 && (data->aux || ismatis)) {
545:       if (!loadedSym) SETERRQ(PETSC_COMM_SELF, PETSC_ERR_PLIB, "HPDDM library not loaded, cannot use more than one level");
546:       MatSetOption(P, MAT_SUBMAT_SINGLEIS, PETSC_TRUE);
547:       if (ismatis) {
548:         /* needed by HPDDM (currently) so that the partition of unity is 0 on subdomain interfaces */
549:         MatIncreaseOverlap(P, 1, is, 1);
550:         ISDestroy(&data->is);
551:         data->is = is[0];
552: #if defined(PETSC_USE_DEBUG)
553:       } else {
554:         PetscBool equal;
555:         IS        intersect;

557:         ISIntersect(data->is, loc, &intersect);
558:         ISEqualUnsorted(loc, intersect, &equal);
559:         ISDestroy(&intersect);
560:         if (!equal) SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG, "IS of the auxiliary Mat does not include all local rows of A");
561: #endif
562:       }
563:       MatCreateSubMatrices(P, 1, is, is, MAT_INITIAL_MATRIX, &sub);
564:       /* Vec holding the partition of unity */
565:       if (!data->levels[0]->D) {
566:         ISGetLocalSize(data->is, &n);
567:         VecCreateMPI(PETSC_COMM_SELF, n, PETSC_DETERMINE, &data->levels[0]->D);
568:       }
569:       if (!data->levels[0]->scatter) {
570:         MatCreateVecs(P, &xin, NULL);
571:         if (ismatis) {
572:           MatDestroy(&P);
573:         }
574:         VecScatterCreate(xin, data->is, data->levels[0]->D, NULL, &data->levels[0]->scatter);
575:         VecDestroy(&xin);
576:       }
577:       if (data->levels[0]->P) {
578:         /* if the pattern is the same and PCSetUp has previously succeeded, reuse HPDDM buffers and connectivity */
579:         HPDDM::Schwarz<PetscScalar>::destroy(data->levels[0], pc->setupcalled < 1 || pc->flag == DIFFERENT_NONZERO_PATTERN ? PETSC_TRUE : PETSC_FALSE);
580:       }
581:       if (!data->levels[0]->P) data->levels[0]->P = new HPDDM::Schwarz<PetscScalar>();
582:       (*loadedSym)(data->levels[0]->P, loc, data->is, sub[0], ismatis ? C : data->aux, initial, &data->N, data->levels);
583:       MatDestroySubMatrices(1, &sub);
584:       if (ismatis) data->is = NULL;
585:       for (n = 0; n < data->N - 1; ++n) {
586:         if (data->levels[n]->P) {
587:           PC spc;

589:           /* force the PC to be PCSHELL to do the coarse grid corrections */
590:           KSPSetSkipPCSetFromOptions(data->levels[n]->ksp, PETSC_TRUE);
591:           KSPGetPC(data->levels[n]->ksp, &spc);
592:           PCSetType(spc, PCSHELL);
593:           PCShellSetContext(spc, data->levels[n]);
594:           PCShellSetSetUp(spc, PCHPDDMShellSetUp);
595:           PCShellSetApply(spc, PCHPDDMShellApply);
596:           PCShellSetDestroy(spc, PCHPDDMShellDestroy);
597:           if (!data->levels[n]->pc) {
598:             PCCreate(PetscObjectComm((PetscObject)data->levels[n]->ksp), &data->levels[n]->pc);
599:           }
600:           PCSetUp(spc);
601:         }
602:       }
603:     } else outer = PETSC_TRUE;
604:     if (!ismatis && subdomains) {
605:       if (outer) {
606:         KSPGetPC(data->levels[0]->ksp, &inner);
607:         PCASMSetLocalSubdomains(inner, 1, is, &loc);
608:       } else {
609:         PCASMSetLocalSubdomains(data->levels[0]->pc, 1, is, &loc);
610:       }
611:     }
612:     ISDestroy(&loc);
613:   } else data->N = 1; /* enforce this value to 1 if there is no way to build another level */
614:   if (requested != data->N) {
615:     PetscInfo4(pc, "%D levels requested, only %D built. Options for level(s) > %D, including -%spc_hpddm_coarse_ will not be taken into account.\n", requested, data->N, data->N, pcpre ? pcpre : "");
616:     PetscInfo2(pc, "It is best to tune parameters, e.g., a higher value for -%spc_hpddm_levels_%D_eps_threshold so that at least one local deflation vector will be selected.\n", pcpre ? pcpre : "", data->N);
617:     /* cannot use PCHPDDMShellDestroy because PCSHELL not set for unassembled levels */
618:     for (n = data->N - 1; n < requested - 1; ++n) {
619:       if (data->levels[n]->P) {
620:         HPDDM::Schwarz<PetscScalar>::destroy(data->levels[n], PETSC_TRUE);
621:         VecDestroyVecs(1, &data->levels[n]->v[0]);
622:         VecDestroyVecs(2, &data->levels[n]->v[1]);
623:         VecDestroy(&data->levels[n]->D);
624:         VecScatterDestroy(&data->levels[n]->scatter);
625:       }
626:     }
627: #if defined(PETSC_USE_DEBUG)
628:     SETERRQ6(PetscObjectComm((PetscObject)pc), PETSC_ERR_ARG_WRONG, "%D levels requested, only %D built. Options for level(s) > %D, including -%spc_hpddm_coarse_ will not be taken into account. It is best to tune parameters, e.g., a higher value for -%spc_hpddm_levels_%D_eps_threshold so that at least one local deflation vector will be selected. If you don't want this to error out, compile --with-debugging=0", requested, data->N, data->N, pcpre ? pcpre : "", pcpre ? pcpre : "", data->N);
629: #endif
630:   }

632:   /* these solvers are created after PCSetFromOptions is called */
633:   if (pc->setfromoptionscalled) {
634:     for (n = 0; n < data->N; ++n) {
635:       if (data->levels[n]->ksp) {
636:         KSPSetFromOptions(data->levels[n]->ksp);
637:       }
638:       if (data->levels[n]->pc) {
639:         PCSetFromOptions(data->levels[n]->pc);
640:       }
641:     }
642:     pc->setfromoptionscalled = 0;
643:   }
644:   return(0);
645: }

713: /*MC
714:      PCHPDDM - Interface with the HPDDM library.

716:    This PC may be used to build multilevel spectral domain decomposition methods based on the GenEO framework [2011, 2019]. It may be viewed as an alternative to spectral AMGe or PCBDDC with adaptive selection of constraints. A chronological bibliography of relevant publications linked with PC available in HPDDM through PCHPDDM may be found below.

718:    The matrix to be preconditioned (Pmat) may be unassembled (MATIS) or assembled (MATMPIAIJ, MATMPIBAIJ, or MATMPISBAIJ). For multilevel preconditioning, when using an assembled Pmat, one must provide an auxiliary local Mat (unassembled local operator for GenEO) using PCHPDDMSetAuxiliaryMat. Calling this routine is not needed when using a MATIS Pmat (assembly done internally using MatConvert).

720:    Options Database Keys:
721: +   -pc_hpddm_define_subdomains <true, false> - on the finest level, calls PCASMSetLocalSubdomains with the IS supplied in PCHPDDMSetAuxiliaryMat (only relevant with an assembled Pmat)
722: -   -pc_hpddm_coarse_correction <type, default=deflated> - determines the PCHPDDMCoarseCorrectionType when calling PCApply

724:    Options for subdomain solvers, subdomain eigensolvers (for computing deflation vectors), and the coarse solver can be set with
725: .vb
726:       -pc_hpddm_levels_%d_pc_
727:       -pc_hpddm_levels_%d_ksp_
728:       -pc_hpddm_levels_%d_eps_
729:       -pc_hpddm_levels_%d_p
730:       -pc_hpddm_levels_%d_mat_type_
731:       -pc_hpddm_coarse_
732:       -pc_hpddm_coarse_p
733:       -pc_hpddm_coarse_mat_type_
734: .ve
735:    e.g., -pc_hpddm_levels_1_sub_pc_type lu -pc_hpddm_levels_1_eps_nev 10 -pc_hpddm_levels_2_p 4 -pc_hpddm_levels_2_sub_pc_type lu -pc_hpddm_levels_2_eps_nev 10 -pc_hpddm_coarse_p 2 -pc_hpddm_coarse_mat_type baij will use 10 deflation vectors per subdomain on the fine "level 1", aggregate the fine subdomains into 4 "level 2" subdomains, then use 10 deflation vectors per subdomain on "level 2", and assemble the coarse matrix (of dimension 4 x 10 = 40) on two processes as a MATMPIBAIJ (default is MATMPISBAIJ).

737:    In order to activate a "level N+1" coarse correction, it is mandatory to call -pc_hpddm_levels_N_eps_nev <nu> or -pc_hpddm_levels_N_eps_threshold <val>. The default -pc_hpddm_coarse_p value is 1, meaning that the coarse operator is aggregated on a single process.

739:    This preconditioner requires that you build PETSc with SLEPc (--download-slepc=1). By default, the underlying concurrent eigenproblems are solved using SLEPc shift-and-invert spectral transformation. This is usually what gives the best performance for GenEO, cf. [2011, 2013]. As stated above, SLEPc options are available through -pc_hpddm_levels_%d_, e.g., -pc_hpddm_levels_1_eps_type arpack -pc_hpddm_levels_1_eps_threshold 0.1 -pc_hpddm_levels_1_st_type sinvert.

741:    References:
742: +   2011 - A robust two-level domain decomposition preconditioner for systems of PDEs. Spillane, Dolean, Hauret, Nataf, Pechstein, and Scheichl. Comptes Rendus Mathematique.
743: .   2013 - Scalable Domain Decomposition Preconditioners For Heterogeneous Elliptic Problems. Jolivet, Hecht, Nataf, and Prud'homme. SC13.
744: .   2015 - An Introduction to Domain Decomposition Methods: Algorithms, Theory, and Parallel Implementation. Dolean, Jolivet, and Nataf. SIAM.
745: -   2019 - A Multilevel Schwarz Preconditioner Based on a Hierarchy of Robust Coarse Spaces. Al Daas, Grigori, Jolivet, and Tournier.

747:    Level: intermediate

749: .seealso:  PCCreate(), PCSetType(), PCType (for list of available types), PC, PCHPDDMSetAuxiliaryMat(), MATIS, PCBDDC, PCDEFLATION, PCTELESCOPE
750: M*/
751: PETSC_EXTERN PetscErrorCode PCCreate_HPDDM(PC pc)
752: {
753:   PetscBool      found;
754:   char           lib[PETSC_MAX_PATH_LEN], dlib[PETSC_MAX_PATH_LEN], dir[PETSC_MAX_PATH_LEN];
755:   PC_HPDDM       *data;

759:   if (!loadedSym) {
760:     PetscStrcpy(dir, "${PETSC_LIB_DIR}");
761:     PetscOptionsGetString(NULL, NULL, "-hpddm_dir", dir, sizeof(dir), NULL);
762:     PetscSNPrintf(lib, sizeof(lib), "%s/libhpddm_petsc", dir);
763:     PetscDLLibraryRetrieve(PETSC_COMM_SELF, lib, dlib, 1024, &found);
764:     if (found) {
765:       PetscDLLibraryAppend(PETSC_COMM_SELF, &PetscDLLibrariesLoaded, dlib);
766:     } else SETERRQ1(PETSC_COMM_SELF, PETSC_ERR_PLIB, "%s not found", lib);
767:     PetscDLLibrarySym(PETSC_COMM_SELF, &PetscDLLibrariesLoaded, NULL, "PCHPDDM_Internal", (void**)&loadedSym);
768:   }
769:   if (!loadedSym) SETERRQ1(PETSC_COMM_SELF, PETSC_ERR_PLIB, "PCHPDDM_Internal symbol not found in %s", lib);
770:   PetscNewLog(pc, &data);
771:   pc->data                     = data;
772:   pc->ops->reset               = PCReset_HPDDM;
773:   pc->ops->destroy             = PCDestroy_HPDDM;
774:   pc->ops->setfromoptions      = PCSetFromOptions_HPDDM;
775:   pc->ops->setup               = PCSetUp_HPDDM;
776:   pc->ops->apply               = PCApply_HPDDM;
777:   pc->ops->view                = PCView_HPDDM;
778:   pc->ops->applytranspose      = 0;
779:   pc->ops->applysymmetricleft  = 0;
780:   pc->ops->applysymmetricright = 0;
781:   PetscObjectComposeFunction((PetscObject)pc, "PCHPDDMSetAuxiliaryMat_C", PCHPDDMSetAuxiliaryMat_HPDDM);
782:   PetscObjectComposeFunction((PetscObject)pc, "PCHPDDMSetCoarseCorrectionType_C", PCHPDDMSetCoarseCorrectionType_HPDDM);
783:   PetscObjectComposeFunction((PetscObject)pc, "PCHPDDMGetCoarseCorrectionType_C", PCHPDDMGetCoarseCorrectionType_HPDDM);
784:   return(0);
785: }
