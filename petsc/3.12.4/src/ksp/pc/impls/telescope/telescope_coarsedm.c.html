124:     void           *dmfine_kspctx = NULL,*dmcoarse_kspctx = NULL;

137:       PetscInfo(pc,"PCTelescope: KSPSetComputeOperators fetched from parent DM\n");
138:       if (PCTelescope_isActiveRank(sred)) {
139:         DMGetApplicationContext(ctx->dm_coarse,&dmcoarse_appctx);
140:         DMShellGetContext(ctx->dm_coarse,&dmcoarse_shellctx);
141:       }

143:       /* Assume that if the fine operator didn't require any context, neither will the coarse */
144:       if (!dmfine_kspctx) {
145:         dmcoarse_kspctx = NULL;
146:         PetscInfo(pc,"PCTelescope: KSPSetComputeOperators using NULL context\n");
147:       } else {

149:         PetscInfo(pc,"PCTelescope: KSPSetComputeOperators detected non-NULL context from parent DM \n");
150:         if (PCTelescope_isActiveRank(sred)) {

152:           if (dmfine_kspctx == dmfine_appctx) {
153:             dmcoarse_kspctx = dmcoarse_appctx;
154:             PetscInfo(pc,"PCTelescope: KSPSetComputeOperators using context from DM->ApplicationContext\n");
155:             if (!dmcoarse_kspctx) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_USER,"Non NULL dmfine->kspctx == dmfine->appctx. NULL dmcoarse->appctx found. Likely this is an error");
156:           } else if (dmfine_kspctx == dmfine_shellctx) {
157:             dmcoarse_kspctx = dmcoarse_shellctx;

159:             if (!dmcoarse_kspctx) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_USER,"Non NULL dmfine->kspctx == dmfine.shell->ctx. NULL dmcoarse.shell->ctx found. Likely this is an error");
160:           }
161:           ctx->dmksp_context_determined = dmcoarse_kspctx;

163:           /* look for user provided method to fetch the context */

165:             PetscErrorCode (*fp_get_coarsedm_context)(DM,void**) = NULL;
166:             void *dmcoarse_context_user = NULL;

169:             PetscSNPrintf(dmcoarse_method,sizeof(dmcoarse_method),"PCTelescopeGetCoarseDMKSPContext");

171:             if (fp_get_coarsedm_context) {
172:               PetscInfo(pc,"PCTelescope: Found composed method PCTelescopeGetCoarseDMKSPContext from coarse DM\n");
173:               fp_get_coarsedm_context(ctx->dm_coarse,&dmcoarse_context_user);
174:               ctx->dmksp_context_user = dmcoarse_context_user;
175:               dmcoarse_kspctx = dmcoarse_context_user;
176:             } else {
177:               PetscInfo(pc,"PCTelescope: Failed to find composed method PCTelescopeGetCoarseDMKSPContext from coarse DM\n");
178:             }
179:           }

181:           if (!dmcoarse_kspctx) {
182:             PetscInfo(pc,"PCTelescope: KSPSetComputeOperators failed to determine the context to use on sub-communicator\n");
183:             SETERRQ(PETSC_COMM_SELF,PETSC_ERR_USER,"Cannot determine which context with use for KSPSetComputeOperators() on sub-communicator");
184:           }
185:         }
186:       }

190:       using_kspcomputeoperators = PETSC_TRUE;

192:       if (PCTelescope_isActiveRank(sred)) {
193:         /* sub ksp inherits dmksp_func and context provided by user */

196:         KSPSetDMActive(sred->ksp,PETSC_TRUE);

198:     }
199:   }

201:   if (!has_perm && has_kspcomputeoperators && !using_kspcomputeoperators) SETERRQ(comm,PETSC_ERR_SUP,"No method to permute an operator was found on the parent DM. A method for KSPSetComputeOperators() was provided but it was requested to be ignored. Telescope setup cannot proceed");
202:   if (!has_perm && !has_kspcomputeoperators) SETERRQ(comm,PETSC_ERR_SUP,"No method to permute an operator was found on the parent DM. No method for KSPSetComputeOperators() was provided. Telescope setup cannot proceed");

204:   {
205:     char dmfine_method[PETSC_MAX_PATH_LEN];

207:     PetscSNPrintf(dmfine_method,sizeof(dmfine_method),"PCTelescopeFieldScatter");

228:   PCTelescopeSetUp_scatters_CoarseDM(pc,sred,ctx);

232: PetscErrorCode PCApply_Telescope_CoarseDM(PC pc,Vec x,Vec y)
233: {
234:   PC_Telescope             sred = (PC_Telescope)pc->data;
235:   PetscErrorCode           ierr;
236:   Vec                      xred,yred;
237:   PC_Telescope_CoarseDMCtx *ctx;

240:   ctx = (PC_Telescope_CoarseDMCtx*)sred->dm_ctx;
241:   xred = sred->xred;
242:   yred = sred->yred;

244:   PetscCitationsRegister(citation,&cited);

246:   if (ctx->fp_dm_state_scatter) {
247:     ctx->fp_dm_state_scatter(ctx->dm_fine,SCATTER_FORWARD,ctx->dm_coarse);
248:   }

250:   ctx->fp_dm_field_scatter(ctx->dm_fine,x,SCATTER_FORWARD,ctx->dm_coarse,xred);

252:   /* solve */
253:   if (PCTelescope_isActiveRank(sred)) {
254:     KSPSolve(sred->ksp,xred,yred);
255:   }

257:   ctx->fp_dm_field_scatter(ctx->dm_fine,y,SCATTER_REVERSE,ctx->dm_coarse,yred);
258:   return(0);
259: }

261: PetscErrorCode PCTelescopeSubNullSpaceCreate_CoarseDM(PC pc,PC_Telescope sred,MatNullSpace nullspace,MatNullSpace *sub_nullspace)
262: {
263:   PetscErrorCode           ierr;
264:   PetscBool                has_const;
265:   PetscInt                 k,n = 0;
266:   const Vec                *vecs;
267:   Vec                      *sub_vecs = NULL;
268:   MPI_Comm                 subcomm;
269:   PC_Telescope_CoarseDMCtx *ctx;

272:   ctx = (PC_Telescope_CoarseDMCtx*)sred->dm_ctx;
273:   subcomm = sred->subcomm;
274:   MatNullSpaceGetVecs(nullspace,&has_const,&n,&vecs);

276:   if (PCTelescope_isActiveRank(sred)) {
277:     /* create new vectors */
278:     if (n) {
279:       VecDuplicateVecs(sred->xred,n,&sub_vecs);
280:     }
281:   }

283:   /* copy entries */
284:   for (k=0; k<n; k++) {
285:     ctx->fp_dm_field_scatter(ctx->dm_fine,vecs[k],SCATTER_FORWARD,ctx->dm_coarse,sub_vecs[k]);
286:   }

288:   if (PCTelescope_isActiveRank(sred)) {
289:     /* create new (near) nullspace for redundant object */
290:     MatNullSpaceCreate(subcomm,has_const,n,sub_vecs,sub_nullspace);
291:     VecDestroyVecs(n,&sub_vecs);
292:   }
293:   return(0);
294: }

296: PetscErrorCode PCTelescopeMatNullSpaceCreate_CoarseDM(PC pc,PC_Telescope sred,Mat sub_mat)
297: {
298:   PetscErrorCode           ierr;
299:   Mat                      B;
300:   PC_Telescope_CoarseDMCtx *ctx;

303:   ctx = (PC_Telescope_CoarseDMCtx*)sred->dm_ctx;
304:   PCGetOperators(pc,NULL,&B);
305:   {
306:     MatNullSpace nullspace,sub_nullspace;
307:     MatGetNullSpace(B,&nullspace);
308:     if (nullspace) {
309:       PetscInfo(pc,"PCTelescope: generating nullspace (CoarseDM)\n");
310:       PCTelescopeSubNullSpaceCreate_CoarseDM(pc,sred,nullspace,&sub_nullspace);

312:       /* attach any user nullspace removal methods and contexts */
313:       if (PCTelescope_isActiveRank(sred)) {
314:         void *context = NULL;
315:         if (nullspace->remove && !nullspace->rmctx){
316:           MatNullSpaceSetFunction(sub_nullspace,nullspace->remove,context);
317:         } else if (nullspace->remove && nullspace->rmctx) {
318:           char           dmcoarse_method[PETSC_MAX_PATH_LEN];
319:           PetscErrorCode (*fp_get_coarsedm_context)(DM,void**) = NULL;

321:           PetscSNPrintf(dmcoarse_method,sizeof(dmcoarse_method),"PCTelescopeGetCoarseDMNullSpaceUserContext");
322:           PetscObjectQueryFunction((PetscObject)ctx->dm_coarse,dmcoarse_method,&fp_get_coarsedm_context);
323:           if (!context) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_SUP,"Propagation of user null-space removal method with non-NULL context requires the coarse DM be composed with a function named \"%s\"",dmcoarse_method);
324:           MatNullSpaceSetFunction(sub_nullspace,nullspace->remove,context);
325:         }
326:       }

328:       if (PCTelescope_isActiveRank(sred)) {
329:         MatSetNullSpace(sub_mat,sub_nullspace);
330:         MatNullSpaceDestroy(&sub_nullspace);
331:       }
332:     }
333:   }
334:   {
335:     MatNullSpace nearnullspace,sub_nearnullspace;
336:     MatGetNearNullSpace(B,&nearnullspace);
337:     if (nearnullspace) {
338:       PetscInfo(pc,"PCTelescope: generating near nullspace (CoarseDM)\n");
339:       PCTelescopeSubNullSpaceCreate_CoarseDM(pc,sred,nearnullspace,&sub_nearnullspace);

341:       /* attach any user nullspace removal methods and contexts */
342:       if (PCTelescope_isActiveRank(sred)) {
343:         void *context = NULL;
344:         if (nearnullspace->remove && !nearnullspace->rmctx){
345:           MatNullSpaceSetFunction(sub_nearnullspace,nearnullspace->remove,context);
346:         } else if (nearnullspace->remove && nearnullspace->rmctx) {
347:           char           dmcoarse_method[PETSC_MAX_PATH_LEN];
348:           PetscErrorCode (*fp_get_coarsedm_context)(DM,void**) = NULL;

350:           PetscSNPrintf(dmcoarse_method,sizeof(dmcoarse_method),"PCTelescopeGetCoarseDMNearNullSpaceUserContext");
351:           PetscObjectQueryFunction((PetscObject)ctx->dm_coarse,dmcoarse_method,&fp_get_coarsedm_context);
352:           if (!context) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_SUP,"Propagation of user near null-space removal method with non-NULL context requires the coarse DM be composed with a function named \"%s\"",dmcoarse_method);
353:           MatNullSpaceSetFunction(sub_nearnullspace,nearnullspace->remove,context);
354:         }
355:       }

357:       if (PCTelescope_isActiveRank(sred)) {
358:         MatSetNearNullSpace(sub_mat,sub_nearnullspace);
359:         MatNullSpaceDestroy(&sub_nearnullspace);
360:       }
361:     }
362:   }
363:   return(0);
364: }
