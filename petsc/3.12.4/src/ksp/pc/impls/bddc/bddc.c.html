1316: /*
1317:    PCPreSolve_BDDC - Changes the right hand side and (if necessary) the initial
1318:                      guess if a transformation of basis approach has been selected.

1320:    Input Parameter:
1321: +  pc - the preconditioner contex

1323:    Application Interface Routine: PCPreSolve()

1325:    Notes:
1326:      The interface routine PCPreSolve() is not usually called directly by
1327:    the user, but instead is called by KSPSolve().
1328: */
1329: static PetscErrorCode PCPreSolve_BDDC(PC pc, KSP ksp, Vec rhs, Vec x)
1330: {
1332:   PC_BDDC        *pcbddc = (PC_BDDC*)pc->data;
1333:   PC_IS          *pcis = (PC_IS*)(pc->data);
1334:   Vec            used_vec;
1335:   PetscBool      iscg = PETSC_FALSE, save_rhs = PETSC_TRUE, benign_correction_computed;

1338:   /* if we are working with CG, one dirichlet solve can be avoided during Krylov iterations */
1339:   if (ksp) {
1340:     PetscBool isgroppcg, ispipecg, ispipelcg, ispipecgrr;

1342:     PetscObjectTypeCompare((PetscObject)ksp,KSPCG,&iscg);
1343:     PetscObjectTypeCompare((PetscObject)ksp,KSPGROPPCG,&isgroppcg);
1344:     PetscObjectTypeCompare((PetscObject)ksp,KSPPIPECG,&ispipecg);
1345:     PetscObjectTypeCompare((PetscObject)ksp,KSPPIPECG,&ispipelcg);
1346:     PetscObjectTypeCompare((PetscObject)ksp,KSPPIPECGRR,&ispipecgrr);
1347:     iscg = (PetscBool)(iscg || isgroppcg || ispipecg || ispipelcg || ispipecgrr);
1348:     if (pcbddc->benign_apply_coarse_only || pcbddc->switch_static || !iscg || pc->mat != pc->pmat) {
1349:       PCBDDCSetUseExactDirichlet(pc,PETSC_FALSE);
1350:     }
1351:   }
1352:   if (pcbddc->benign_apply_coarse_only || pcbddc->switch_static || pc->mat != pc->pmat) {
1353:     PCBDDCSetUseExactDirichlet(pc,PETSC_FALSE);
1354:   }

1356:   /* Creates parallel work vectors used in presolve */
1357:   if (!pcbddc->original_rhs) {
1358:     VecDuplicate(pcis->vec1_global,&pcbddc->original_rhs);
1359:   }
1360:   if (!pcbddc->temp_solution) {
1361:     VecDuplicate(pcis->vec1_global,&pcbddc->temp_solution);
1362:   }

1364:   pcbddc->temp_solution_used = PETSC_FALSE;
1365:   if (x) {
1366:     PetscObjectReference((PetscObject)x);
1367:     used_vec = x;
1368:   } else { /* it can only happen when calling PCBDDCMatFETIDPGetRHS */
1369:     PetscObjectReference((PetscObject)pcbddc->temp_solution);
1370:     used_vec = pcbddc->temp_solution;
1371:     VecSet(used_vec,0.0);
1372:     pcbddc->temp_solution_used = PETSC_TRUE;
1373:     VecCopy(rhs,pcbddc->original_rhs);
1374:     save_rhs = PETSC_FALSE;
1375:     pcbddc->eliminate_dirdofs = PETSC_TRUE;
1376:   }

1378:   /* hack into ksp data structure since PCPreSolve comes earlier than setting to zero the guess in src/ksp/ksp/interface/itfunc.c */
1379:   if (ksp) {
1380:     /* store the flag for the initial guess since it will be restored back during PCPostSolve_BDDC */
1381:     KSPGetInitialGuessNonzero(ksp,&pcbddc->ksp_guess_nonzero);
1382:     if (!pcbddc->ksp_guess_nonzero) {
1383:       VecSet(used_vec,0.0);
1384:     }
1385:   }

1387:   pcbddc->rhs_change = PETSC_FALSE;
1388:   /* Take into account zeroed rows -> change rhs and store solution removed */
1389:   if (rhs && pcbddc->eliminate_dirdofs) {
1390:     IS dirIS = NULL;

1392:     /* DirichletBoundariesLocal may not be consistent among neighbours; gets a dirichlet dofs IS from graph (may be cached) */
1393:     PCBDDCGraphGetDirichletDofs(pcbddc->mat_graph,&dirIS);
1394:     if (dirIS) {
1395:       Mat_IS            *matis = (Mat_IS*)pc->pmat->data;
1396:       PetscInt          dirsize,i,*is_indices;
1397:       PetscScalar       *array_x;
1398:       const PetscScalar *array_diagonal;

1400:       MatGetDiagonal(pc->pmat,pcis->vec1_global);
1401:       VecPointwiseDivide(pcis->vec1_global,rhs,pcis->vec1_global);
1402:       VecScatterBegin(matis->rctx,pcis->vec1_global,pcis->vec2_N,INSERT_VALUES,SCATTER_FORWARD);
1403:       VecScatterEnd(matis->rctx,pcis->vec1_global,pcis->vec2_N,INSERT_VALUES,SCATTER_FORWARD);
1404:       VecScatterBegin(matis->rctx,used_vec,pcis->vec1_N,INSERT_VALUES,SCATTER_FORWARD);
1405:       VecScatterEnd(matis->rctx,used_vec,pcis->vec1_N,INSERT_VALUES,SCATTER_FORWARD);
1406:       ISGetLocalSize(dirIS,&dirsize);
1407:       VecGetArray(pcis->vec1_N,&array_x);
1408:       VecGetArrayRead(pcis->vec2_N,&array_diagonal);
1409:       ISGetIndices(dirIS,(const PetscInt**)&is_indices);
1410:       for (i=0; i<dirsize; i++) array_x[is_indices[i]] = array_diagonal[is_indices[i]];
1411:       ISRestoreIndices(dirIS,(const PetscInt**)&is_indices);
1412:       VecRestoreArrayRead(pcis->vec2_N,&array_diagonal);
1413:       VecRestoreArray(pcis->vec1_N,&array_x);
1414:       VecScatterBegin(matis->rctx,pcis->vec1_N,used_vec,INSERT_VALUES,SCATTER_REVERSE);
1415:       VecScatterEnd(matis->rctx,pcis->vec1_N,used_vec,INSERT_VALUES,SCATTER_REVERSE);
1416:       pcbddc->rhs_change = PETSC_TRUE;
1417:       ISDestroy(&dirIS);
1418:     }
1419:   }

1421:   /* remove the computed solution or the initial guess from the rhs */
1422:   if (pcbddc->rhs_change || (ksp && pcbddc->ksp_guess_nonzero) ) {
1423:     /* save the original rhs */
1424:     if (save_rhs) {
1425:       VecSwap(rhs,pcbddc->original_rhs);
1426:       save_rhs = PETSC_FALSE;
1427:     }
1428:     pcbddc->rhs_change = PETSC_TRUE;
1429:     VecScale(used_vec,-1.0);
1430:     MatMultAdd(pc->mat,used_vec,pcbddc->original_rhs,rhs);
1431:     VecScale(used_vec,-1.0);
1432:     VecCopy(used_vec,pcbddc->temp_solution);
1433:     pcbddc->temp_solution_used = PETSC_TRUE;
1434:     if (ksp) {
1435:       KSPSetInitialGuessNonzero(ksp,PETSC_FALSE);
1436:     }
1437:   }
1438:   VecDestroy(&used_vec);

1440:   /* compute initial vector in benign space if needed
1441:      and remove non-benign solution from the rhs */
1442:   benign_correction_computed = PETSC_FALSE;
1443:   if (rhs && pcbddc->benign_compute_correction && (pcbddc->benign_have_null || pcbddc->benign_apply_coarse_only)) {
1444:     /* compute u^*_h using ideas similar to those in Xuemin Tu's PhD thesis (see Section 4.8.1)
1445:        Recursively apply BDDC in the multilevel case */
1446:     if (!pcbddc->benign_vec) {
1447:       VecDuplicate(rhs,&pcbddc->benign_vec);
1448:     }
1449:     /* keep applying coarse solver unless we no longer have benign subdomains */
1450:     pcbddc->benign_apply_coarse_only = pcbddc->benign_have_null ? PETSC_TRUE : PETSC_FALSE;
1451:     if (!pcbddc->benign_skip_correction) {
1452:       PCApply_BDDC(pc,rhs,pcbddc->benign_vec);
1453:       benign_correction_computed = PETSC_TRUE;
1454:       if (pcbddc->temp_solution_used) {
1455:         VecAXPY(pcbddc->temp_solution,1.0,pcbddc->benign_vec);
1456:       }
1457:       VecScale(pcbddc->benign_vec,-1.0);
1458:       /* store the original rhs if not done earlier */
1459:       if (save_rhs) {
1460:         VecSwap(rhs,pcbddc->original_rhs);
1461:       }
1462:       if (pcbddc->rhs_change) {
1463:         MatMultAdd(pc->mat,pcbddc->benign_vec,rhs,rhs);
1464:       } else {
1465:         MatMultAdd(pc->mat,pcbddc->benign_vec,pcbddc->original_rhs,rhs);
1466:       }
1467:       pcbddc->rhs_change = PETSC_TRUE;
1468:     }
1469:     pcbddc->benign_apply_coarse_only = PETSC_FALSE;
1470:   } else {
1471:     VecDestroy(&pcbddc->benign_vec);
1472:   }

1474:   /* dbg output */
1475:   if (pcbddc->dbg_flag && benign_correction_computed) {
1476:     Vec v;

1478:     VecDuplicate(pcis->vec1_global,&v);
1479:     if (pcbddc->ChangeOfBasisMatrix) {
1480:       MatMultTranspose(pcbddc->ChangeOfBasisMatrix,rhs,v);
1481:     } else {
1482:       VecCopy(rhs,v);
1483:     }
1484:     PCBDDCBenignGetOrSetP0(pc,v,PETSC_TRUE);
1485:     PetscViewerASCIIPrintf(pcbddc->dbg_viewer,"LEVEL %D: is the correction benign?\n",pcbddc->current_level);
1486:     PetscScalarView(pcbddc->benign_n,pcbddc->benign_p0,pcbddc->dbg_viewer);
1487:     PetscViewerFlush(pcbddc->dbg_viewer);
1488:     VecDestroy(&v);
1489:   }

1491:   /* set initial guess if using PCG */
1492:   pcbddc->exact_dirichlet_trick_app = PETSC_FALSE;
1493:   if (x && pcbddc->use_exact_dirichlet_trick) {
1494:     VecSet(x,0.0);
1495:     if (pcbddc->ChangeOfBasisMatrix && pcbddc->change_interior) {
1496:       if (benign_correction_computed) { /* we have already saved the changed rhs */
1497:         VecLockReadPop(pcis->vec1_global);
1498:       } else {
1499:         MatMultTranspose(pcbddc->ChangeOfBasisMatrix,rhs,pcis->vec1_global);
1500:       }
1501:       VecScatterBegin(pcis->global_to_D,pcis->vec1_global,pcis->vec1_D,INSERT_VALUES,SCATTER_FORWARD);
1502:       VecScatterEnd(pcis->global_to_D,pcis->vec1_global,pcis->vec1_D,INSERT_VALUES,SCATTER_FORWARD);
1503:     } else {
1504:       VecScatterBegin(pcis->global_to_D,rhs,pcis->vec1_D,INSERT_VALUES,SCATTER_FORWARD);
1505:       VecScatterEnd(pcis->global_to_D,rhs,pcis->vec1_D,INSERT_VALUES,SCATTER_FORWARD);
1506:     }
1507:     KSPSolve(pcbddc->ksp_D,pcis->vec1_D,pcis->vec2_D);
1508:     KSPCheckSolve(pcbddc->ksp_D,pc,pcis->vec2_D);
1509:     if (pcbddc->ChangeOfBasisMatrix && pcbddc->change_interior) {
1510:       VecSet(pcis->vec1_global,0.);
1511:       VecScatterBegin(pcis->global_to_D,pcis->vec2_D,pcis->vec1_global,INSERT_VALUES,SCATTER_REVERSE);
1512:       VecScatterEnd(pcis->global_to_D,pcis->vec2_D,pcis->vec1_global,INSERT_VALUES,SCATTER_REVERSE);
1513:       MatMult(pcbddc->ChangeOfBasisMatrix,pcis->vec1_global,x);
1514:     } else {
1515:       VecScatterBegin(pcis->global_to_D,pcis->vec2_D,x,INSERT_VALUES,SCATTER_REVERSE);
1516:       VecScatterEnd(pcis->global_to_D,pcis->vec2_D,x,INSERT_VALUES,SCATTER_REVERSE);
1517:     }
1518:     if (ksp) {
1519:       KSPSetInitialGuessNonzero(ksp,PETSC_TRUE);
1520:     }
1521:     pcbddc->exact_dirichlet_trick_app = PETSC_TRUE;
1522:   } else if (pcbddc->ChangeOfBasisMatrix && pcbddc->change_interior && benign_correction_computed && pcbddc->use_exact_dirichlet_trick) {
1523:     VecLockReadPop(pcis->vec1_global);
1524:   }
1525:   return(0);
1526: }

1528: /*
1529:    PCPostSolve_BDDC - Changes the computed solution if a transformation of basis
1530:                      approach has been selected. Also, restores rhs to its original state.

1532:    Input Parameter:
1533: +  pc - the preconditioner contex

1535:    Application Interface Routine: PCPostSolve()

1537:    Notes:
1538:      The interface routine PCPostSolve() is not usually called directly by
1539:      the user, but instead is called by KSPSolve().
1540: */
1541: static PetscErrorCode PCPostSolve_BDDC(PC pc, KSP ksp, Vec rhs, Vec x)
1542: {
1544:   PC_BDDC        *pcbddc = (PC_BDDC*)pc->data;

1547:   /* add solution removed in presolve */
1548:   if (x && pcbddc->rhs_change) {
1549:     if (pcbddc->temp_solution_used) {
1550:       VecAXPY(x,1.0,pcbddc->temp_solution);
1551:     } else if (pcbddc->benign_compute_correction && pcbddc->benign_vec) {
1552:       VecAXPY(x,-1.0,pcbddc->benign_vec);
1553:     }
1554:     /* restore to original state (not for FETI-DP) */
1555:     if (ksp) pcbddc->temp_solution_used = PETSC_FALSE;
1556:   }

1558:   /* restore rhs to its original state (not needed for FETI-DP) */
1559:   if (rhs && pcbddc->rhs_change) {
1560:     VecSwap(rhs,pcbddc->original_rhs);
1561:     pcbddc->rhs_change = PETSC_FALSE;
1562:   }
1563:   /* restore ksp guess state */
1564:   if (ksp) {
1565:     KSPSetInitialGuessNonzero(ksp,pcbddc->ksp_guess_nonzero);
1566:     /* reset flag for exact dirichlet trick */
1567:     pcbddc->exact_dirichlet_trick_app = PETSC_FALSE;
1568:   }
1569:   return(0);
1570: }

1576:    Input Parameter:
1577: +  pc - the preconditioner context

1579:    Application Interface Routine: PCSetUp()

1581:    Notes:
1582:      The interface routine PCSetUp() is not usually called directly by
1583:      the user, but instead is called by PCApply() if necessary.

1587:   PC_BDDC*        pcbddc = (PC_BDDC*)pc->data;
1588:   PCBDDCSubSchurs sub_schurs;
1589:   Mat_IS*         matis;
1590:   MatNullSpace    nearnullspace;
1591:   Mat             lA;
1592:   IS              lP,zerodiag = NULL;
1593:   PetscInt        nrows,ncols;
1594:   PetscMPIInt     size;
1595:   PetscBool       computesubschurs;
1596:   PetscBool       computeconstraintsmatrix;
1597:   PetscBool       new_nearnullspace_provided,ismatis,rl;
1598:   PetscErrorCode  ierr;

1601:   PetscObjectTypeCompare((PetscObject)pc->pmat,MATIS,&ismatis);

1605:   MPI_Comm_size(PetscObjectComm((PetscObject)pc),&size);

1607:   matis = (Mat_IS*)pc->pmat->data;
1608:   /* the following lines of code should be replaced by a better logic between PCIS, PCNN, PCBDDC and other future nonoverlapping preconditioners */
1609:   /* For BDDC we need to define a local "Neumann" problem different to that defined in PCISSetup
1610:      Also, BDDC builds its own KSP for the Dirichlet problem */
1611:   rl = pcbddc->recompute_topography;
1612:   if (!pc->setupcalled || pc->flag == DIFFERENT_NONZERO_PATTERN) rl = PETSC_TRUE;
1613:   MPIU_Allreduce(&rl,&pcbddc->recompute_topography,1,MPIU_BOOL,MPI_LOR,PetscObjectComm((PetscObject)pc));
1614:   if (pcbddc->recompute_topography) {
1615:     pcbddc->graphanalyzed    = PETSC_FALSE;
1616:     computeconstraintsmatrix = PETSC_TRUE;
1617:   } else {
1618:     computeconstraintsmatrix = PETSC_FALSE;
1619:   }

1621:   /* check parameters' compatibility */
1622:   if (!pcbddc->use_deluxe_scaling) pcbddc->deluxe_zerorows = PETSC_FALSE;
1623:   pcbddc->adaptive_selection   = (PetscBool)(pcbddc->adaptive_threshold[0] != 0.0 || pcbddc->adaptive_threshold[1] != 0.0);
1624:   pcbddc->use_deluxe_scaling   = (PetscBool)(pcbddc->use_deluxe_scaling && size > 1);
1625:   pcbddc->adaptive_selection   = (PetscBool)(pcbddc->adaptive_selection && size > 1);
1626:   pcbddc->adaptive_userdefined = (PetscBool)(pcbddc->adaptive_selection && pcbddc->adaptive_userdefined);
1627:   if (pcbddc->adaptive_selection) pcbddc->use_faces = PETSC_TRUE;

1629:   computesubschurs = (PetscBool)(pcbddc->adaptive_selection || pcbddc->use_deluxe_scaling);

1631:   /* activate all connected components if the netflux has been requested */
1632:   if (pcbddc->compute_nonetflux) {
1633:     pcbddc->use_vertices = PETSC_TRUE;
1634:     pcbddc->use_edges    = PETSC_TRUE;
1635:     pcbddc->use_faces    = PETSC_TRUE;
1636:   }

1638:   /* Get stdout for dbg */
1639:   if (pcbddc->dbg_flag) {
1640:     if (!pcbddc->dbg_viewer) {
1641:       pcbddc->dbg_viewer = PETSC_VIEWER_STDOUT_(PetscObjectComm((PetscObject)pc));
1642:     }
1643:     PetscViewerASCIIPushSynchronized(pcbddc->dbg_viewer);
1644:     PetscViewerASCIIAddTab(pcbddc->dbg_viewer,2*pcbddc->current_level);
1645:   }

1647:   /* process topology information */
1648:   PetscLogEventBegin(PC_BDDC_Topology[pcbddc->current_level],pc,0,0,0);
1649:   if (pcbddc->recompute_topography) {
1650:     PCBDDCComputeLocalTopologyInfo(pc);
1651:     if (pcbddc->discretegradient) {
1652:       PCBDDCNedelecSupport(pc);
1653:     }
1654:   }
1655:   if (pcbddc->corner_selected) pcbddc->use_vertices = PETSC_TRUE;

1657:   /* change basis if requested by the user */
1658:   if (pcbddc->user_ChangeOfBasisMatrix) {
1659:     /* use_change_of_basis flag is used to automatically compute a change of basis from constraints */
1660:     pcbddc->use_change_of_basis = PETSC_FALSE;
1661:     PCBDDCComputeLocalMatrix(pc,pcbddc->user_ChangeOfBasisMatrix);
1662:   } else {
1663:     MatDestroy(&pcbddc->local_mat);
1664:     PetscObjectReference((PetscObject)matis->A);
1665:     pcbddc->local_mat = matis->A;
1666:   }

1668:   /*
1669:      Compute change of basis on local pressures (aka zerodiag dofs) with the benign trick
1670:      This should come earlier then PCISSetUp for extracting the correct subdomain matrices
1671:   */
1672:   PCBDDCBenignShellMat(pc,PETSC_TRUE);
1673:   if (pcbddc->benign_saddle_point) {
1674:     PC_IS* pcis = (PC_IS*)pc->data;

1676:     if (pcbddc->user_ChangeOfBasisMatrix || pcbddc->use_change_of_basis || !computesubschurs) pcbddc->benign_change_explicit = PETSC_TRUE;
1677:     /* detect local saddle point and change the basis in pcbddc->local_mat */
1678:     PCBDDCBenignDetectSaddlePoint(pc,(PetscBool)(!pcbddc->recompute_topography),&zerodiag);
1679:     /* pop B0 mat from local mat */
1680:     PCBDDCBenignPopOrPushB0(pc,PETSC_TRUE);
1681:     /* give pcis a hint to not reuse submatrices during PCISCreate */
1682:     if (pc->flag == SAME_NONZERO_PATTERN && pcis->reusesubmatrices == PETSC_TRUE) {
1683:       if (pcbddc->benign_n && (pcbddc->benign_change_explicit || pcbddc->dbg_flag)) {
1684:         pcis->reusesubmatrices = PETSC_FALSE;
1685:       } else {
1686:         pcis->reusesubmatrices = PETSC_TRUE;
1687:       }
1688:     } else {
1689:       pcis->reusesubmatrices = PETSC_FALSE;
1690:     }
1691:   }

1693:   /* propagate relevant information */
1694:   if (matis->A->symmetric_set) {
1695:     MatSetOption(pcbddc->local_mat,MAT_SYMMETRIC,matis->A->symmetric);
1696:   }
1697:   if (matis->A->spd_set) {
1698:     MatSetOption(pcbddc->local_mat,MAT_SPD,matis->A->spd);
1699:   }

1701:   /* Set up all the "iterative substructuring" common block without computing solvers */
1702:   {
1703:     Mat temp_mat;

1705:     temp_mat = matis->A;
1706:     matis->A = pcbddc->local_mat;
1707:     PCISSetUp(pc,PETSC_TRUE,PETSC_FALSE);
1708:     pcbddc->local_mat = matis->A;
1709:     matis->A = temp_mat;
1710:   }

1712:   /* Analyze interface */
1713:   if (!pcbddc->graphanalyzed) {
1714:     PCBDDCAnalyzeInterface(pc);
1715:     computeconstraintsmatrix = PETSC_TRUE;
1716:     if (pcbddc->adaptive_selection && !pcbddc->use_deluxe_scaling && !pcbddc->mat_graph->twodim) {
1717:       SETERRQ(PetscObjectComm((PetscObject)pc),PETSC_ERR_SUP,"Cannot compute the adaptive primal space for a problem with 3D edges without deluxe scaling");
1718:     }
1719:     if (pcbddc->compute_nonetflux) {
1720:       MatNullSpace nnfnnsp;

1722:       if (!pcbddc->divudotp) SETERRQ(PetscObjectComm((PetscObject)pc),PETSC_ERR_SUP,"Missing divudotp operator");
1723:       PCBDDCComputeNoNetFlux(pc->pmat,pcbddc->divudotp,pcbddc->divudotp_trans,pcbddc->divudotp_vl2l,pcbddc->mat_graph,&nnfnnsp);
1724:       /* TODO what if a nearnullspace is already attached? */
1725:       if (nnfnnsp) {
1726:         MatSetNearNullSpace(pc->pmat,nnfnnsp);
1727:         MatNullSpaceDestroy(&nnfnnsp);
1728:       }
1729:     }
1730:   }
1731:   PetscLogEventEnd(PC_BDDC_Topology[pcbddc->current_level],pc,0,0,0);

1733:   /* check existence of a divergence free extension, i.e.
1734:      b(v_I,p_0) = 0 for all v_I (raise error if not).
1735:      Also, check that PCBDDCBenignGetOrSetP0 works */
1736:   if (pcbddc->benign_saddle_point && pcbddc->dbg_flag > 1) {
1737:     PCBDDCBenignCheck(pc,zerodiag);
1738:   }
1739:   ISDestroy(&zerodiag);

1741:   /* Setup local dirichlet solver ksp_D and sub_schurs solvers */
1742:   if (computesubschurs && pcbddc->recompute_topography) {
1743:     PCBDDCInitSubSchurs(pc);
1744:   }
1745:   /* SetUp Scaling operator (scaling matrices could be needed in SubSchursSetUp)*/
1746:   if (!pcbddc->use_deluxe_scaling) {
1747:     PCBDDCScalingSetUp(pc);
1748:   }

1750:   /* finish setup solvers and do adaptive selection of constraints */
1751:   sub_schurs = pcbddc->sub_schurs;
1752:   if (sub_schurs && sub_schurs->schur_explicit) {
1753:     if (computesubschurs) {
1754:       PCBDDCSetUpSubSchurs(pc);
1755:     }
1756:     PCBDDCSetUpLocalSolvers(pc,PETSC_TRUE,PETSC_FALSE);
1757:   } else {
1758:     PCBDDCSetUpLocalSolvers(pc,PETSC_TRUE,PETSC_FALSE);
1759:     if (computesubschurs) {
1760:       PCBDDCSetUpSubSchurs(pc);
1761:     }
1762:   }
1763:   if (pcbddc->adaptive_selection) {
1764:     PCBDDCAdaptiveSelection(pc);
1765:     computeconstraintsmatrix = PETSC_TRUE;
1766:   }

1768:   /* infer if NullSpace object attached to Mat via MatSetNearNullSpace has changed */
1769:   new_nearnullspace_provided = PETSC_FALSE;
1770:   MatGetNearNullSpace(pc->pmat,&nearnullspace);
1771:   if (pcbddc->onearnullspace) { /* already used nearnullspace */
1772:     if (!nearnullspace) { /* near null space attached to mat has been destroyed */
1773:       new_nearnullspace_provided = PETSC_TRUE;
1774:     } else {
1775:       /* determine if the two nullspaces are different (should be lightweight) */
1776:       if (nearnullspace != pcbddc->onearnullspace) {
1777:         new_nearnullspace_provided = PETSC_TRUE;
1778:       } else { /* maybe the user has changed the content of the nearnullspace so check vectors ObjectStateId */
1779:         PetscInt         i;
1780:         const Vec        *nearnullvecs;
1781:         PetscObjectState state;
1782:         PetscInt         nnsp_size;
1783:         MatNullSpaceGetVecs(nearnullspace,NULL,&nnsp_size,&nearnullvecs);
1784:         for (i=0;i<nnsp_size;i++) {
1785:           PetscObjectStateGet((PetscObject)nearnullvecs[i],&state);
1786:           if (pcbddc->onearnullvecs_state[i] != state) {
1787:             new_nearnullspace_provided = PETSC_TRUE;
1788:             break;
1789:           }
1790:         }
1791:       }
1792:     }
1793:   } else {
1794:     if (!nearnullspace) { /* both nearnullspaces are null */
1795:       new_nearnullspace_provided = PETSC_FALSE;
1796:     } else { /* nearnullspace attached later */
1797:       new_nearnullspace_provided = PETSC_TRUE;
1798:     }
1799:   }

1801:   /* Setup constraints and related work vectors */
1802:   /* reset primal space flags */
1803:   PetscLogEventBegin(PC_BDDC_LocalWork[pcbddc->current_level],pc,0,0,0);
1804:   pcbddc->new_primal_space = PETSC_FALSE;
1805:   pcbddc->new_primal_space_local = PETSC_FALSE;
1806:   if (computeconstraintsmatrix || new_nearnullspace_provided) {
1807:     /* It also sets the primal space flags */
1808:     PCBDDCConstraintsSetUp(pc);
1809:   }
1810:   /* Allocate needed local vectors (which depends on quantities defined during ConstraintsSetUp) */
1811:   PCBDDCSetUpLocalWorkVectors(pc);

1813:   if (pcbddc->use_change_of_basis) {
1814:     PC_IS *pcis = (PC_IS*)(pc->data);

1816:     PCBDDCComputeLocalMatrix(pc,pcbddc->ChangeOfBasisMatrix);
1817:     if (pcbddc->benign_change) {
1818:       MatDestroy(&pcbddc->benign_B0);
1819:       /* pop B0 from pcbddc->local_mat */
1820:       PCBDDCBenignPopOrPushB0(pc,PETSC_TRUE);
1821:     }
1822:     /* get submatrices */
1823:     MatDestroy(&pcis->A_IB);
1824:     MatDestroy(&pcis->A_BI);
1825:     MatDestroy(&pcis->A_BB);
1826:     MatCreateSubMatrix(pcbddc->local_mat,pcis->is_B_local,pcis->is_B_local,MAT_INITIAL_MATRIX,&pcis->A_BB);
1827:     MatCreateSubMatrix(pcbddc->local_mat,pcis->is_I_local,pcis->is_B_local,MAT_INITIAL_MATRIX,&pcis->A_IB);
1828:     MatCreateSubMatrix(pcbddc->local_mat,pcis->is_B_local,pcis->is_I_local,MAT_INITIAL_MATRIX,&pcis->A_BI);
1829:     /* set flag in pcis to not reuse submatrices during PCISCreate */
1830:     pcis->reusesubmatrices = PETSC_FALSE;
1831:   } else if (!pcbddc->user_ChangeOfBasisMatrix && !pcbddc->benign_change) {
1832:     MatDestroy(&pcbddc->local_mat);
1833:     PetscObjectReference((PetscObject)matis->A);
1834:     pcbddc->local_mat = matis->A;
1835:   }

1837:   /* interface pressure block row for B_C */
1838:   PetscObjectQuery((PetscObject)pc,"__KSPFETIDP_lP" ,(PetscObject*)&lP);
1839:   PetscObjectQuery((PetscObject)pc,"__KSPFETIDP_lA" ,(PetscObject*)&lA);
1840:   if (lA && lP) {
1841:     PC_IS*    pcis = (PC_IS*)pc->data;
1842:     Mat       B_BI,B_BB,Bt_BI,Bt_BB;
1843:     PetscBool issym;
1844:     MatIsSymmetric(lA,PETSC_SMALL,&issym);
1845:     if (issym) {
1846:       MatCreateSubMatrix(lA,lP,pcis->is_I_local,MAT_INITIAL_MATRIX,&B_BI);
1847:       MatCreateSubMatrix(lA,lP,pcis->is_B_local,MAT_INITIAL_MATRIX,&B_BB);
1848:       MatCreateTranspose(B_BI,&Bt_BI);
1849:       MatCreateTranspose(B_BB,&Bt_BB);
1850:     } else {
1851:       MatCreateSubMatrix(lA,lP,pcis->is_I_local,MAT_INITIAL_MATRIX,&B_BI);
1852:       MatCreateSubMatrix(lA,lP,pcis->is_B_local,MAT_INITIAL_MATRIX,&B_BB);
1853:       MatCreateSubMatrix(lA,pcis->is_I_local,lP,MAT_INITIAL_MATRIX,&Bt_BI);
1854:       MatCreateSubMatrix(lA,pcis->is_B_local,lP,MAT_INITIAL_MATRIX,&Bt_BB);
1855:     }
1856:     PetscObjectCompose((PetscObject)pc,"__KSPFETIDP_B_BI",(PetscObject)B_BI);
1857:     PetscObjectCompose((PetscObject)pc,"__KSPFETIDP_B_BB",(PetscObject)B_BB);
1858:     PetscObjectCompose((PetscObject)pc,"__KSPFETIDP_Bt_BI",(PetscObject)Bt_BI);
1859:     PetscObjectCompose((PetscObject)pc,"__KSPFETIDP_Bt_BB",(PetscObject)Bt_BB);
1860:     MatDestroy(&B_BI);
1861:     MatDestroy(&B_BB);
1862:     MatDestroy(&Bt_BI);
1863:     MatDestroy(&Bt_BB);
1864:   }
1865:   PetscLogEventEnd(PC_BDDC_LocalWork[pcbddc->current_level],pc,0,0,0);

1867:   /* SetUp coarse and local Neumann solvers */
1868:   PCBDDCSetUpSolvers(pc);
1869:   /* SetUp Scaling operator */
1870:   if (pcbddc->use_deluxe_scaling) {
1871:     PCBDDCScalingSetUp(pc);
1872:   }

1874:   /* mark topography as done */
1875:   pcbddc->recompute_topography = PETSC_FALSE;

1877:   /* wrap pcis->A_IB and pcis->A_BI if we did not change explicitly the variables on the pressures */
1878:   PCBDDCBenignShellMat(pc,PETSC_FALSE);

1880:   if (pcbddc->dbg_flag) {
1881:     PetscViewerASCIISubtractTab(pcbddc->dbg_viewer,2*pcbddc->current_level);
1882:     PetscViewerASCIIPopSynchronized(pcbddc->dbg_viewer);
1883:   }
1884:   return(0);
1885: }

1887: /*
1888:    PCApply_BDDC - Applies the BDDC operator to a vector.

1890:    Input Parameters:
1891: +  pc - the preconditioner context
1892: -  r - input vector (global)

1894:    Output Parameter:
1895: .  z - output vector (global)

1897:    Application Interface Routine: PCApply()
1898:  */
1899: PetscErrorCode PCApply_BDDC(PC pc,Vec r,Vec z)
1900: {
1901:   PC_IS             *pcis = (PC_IS*)(pc->data);
1902:   PC_BDDC           *pcbddc = (PC_BDDC*)(pc->data);
1903:   Mat               lA = NULL;
1904:   PetscInt          n_B = pcis->n_B, n_D = pcis->n - n_B;
1905:   PetscErrorCode    ierr;
1906:   const PetscScalar one = 1.0;
1907:   const PetscScalar m_one = -1.0;
1908:   const PetscScalar zero = 0.0;
1909: /* This code is similar to that provided in nn.c for PCNN
1910:    NN interface preconditioner changed to BDDC
1911:    Added support for M_3 preconditioner in the reference article (code is active if pcbddc->switch_static == PETSC_TRUE) */

1914:   PetscCitationsRegister(citation,&cited);
1915:   if (pcbddc->switch_static) {
1916:     MatISGetLocalMat(pc->useAmat ? pc->mat : pc->pmat,&lA);
1917:   }

1919:   if (pcbddc->ChangeOfBasisMatrix) {
1920:     Vec swap;

1922:     MatMultTranspose(pcbddc->ChangeOfBasisMatrix,r,pcbddc->work_change);
1923:     swap = pcbddc->work_change;
1924:     pcbddc->work_change = r;
1925:     r = swap;
1926:     /* save rhs so that we don't need to apply the change of basis for the exact dirichlet trick in PreSolve */
1927:     if (pcbddc->benign_apply_coarse_only && pcbddc->use_exact_dirichlet_trick && pcbddc->change_interior) {
1928:       VecCopy(r,pcis->vec1_global);
1929:       VecLockReadPush(pcis->vec1_global);
1930:     }
1931:   }
1932:   if (pcbddc->benign_have_null) { /* get p0 from r */
1933:     PCBDDCBenignGetOrSetP0(pc,r,PETSC_TRUE);
1934:   }
1935:   if (pcbddc->interface_extension == PC_BDDC_INTERFACE_EXT_DIRICHLET && !pcbddc->exact_dirichlet_trick_app && !pcbddc->benign_apply_coarse_only) {
1936:     VecCopy(r,z);
1937:     /* First Dirichlet solve */
1938:     VecScatterBegin(pcis->global_to_D,r,pcis->vec1_D,INSERT_VALUES,SCATTER_FORWARD);
1939:     VecScatterEnd(pcis->global_to_D,r,pcis->vec1_D,INSERT_VALUES,SCATTER_FORWARD);
1940:     /*
1941:       Assembling right hand side for BDDC operator
1942:       - pcis->vec1_D for the Dirichlet part (if needed, i.e. pcbddc->switch_static == PETSC_TRUE)
1943:       - pcis->vec1_B the interface part of the global vector z
1944:     */
1945:     if (n_D) {
1946:       KSPSolve(pcbddc->ksp_D,pcis->vec1_D,pcis->vec2_D);
1947:       KSPCheckSolve(pcbddc->ksp_D,pc,pcis->vec2_D);
1948:       VecScale(pcis->vec2_D,m_one);
1949:       if (pcbddc->switch_static) {
1950:         VecSet(pcis->vec1_N,0.);
1951:         VecScatterBegin(pcis->N_to_D,pcis->vec2_D,pcis->vec1_N,INSERT_VALUES,SCATTER_REVERSE);
1952:         VecScatterEnd(pcis->N_to_D,pcis->vec2_D,pcis->vec1_N,INSERT_VALUES,SCATTER_REVERSE);
1953:         if (!pcbddc->switch_static_change) {
1954:           MatMult(lA,pcis->vec1_N,pcis->vec2_N);
1955:         } else {
1956:           MatMult(pcbddc->switch_static_change,pcis->vec1_N,pcis->vec2_N);
1957:           MatMult(lA,pcis->vec2_N,pcis->vec1_N);
1958:           MatMultTranspose(pcbddc->switch_static_change,pcis->vec1_N,pcis->vec2_N);
1959:         }
1960:         VecScatterBegin(pcis->N_to_D,pcis->vec2_N,pcis->vec1_D,ADD_VALUES,SCATTER_FORWARD);

2344:   if (!pcbddc->original_rhs) {
2345:     VecDuplicate(pcis->vec1_global,&pcbddc->original_rhs);
2346:   }
2347:   if (mat_ctx->rhs_flip) {
2348:     VecPointwiseMult(pcbddc->original_rhs,standard_rhs,mat_ctx->rhs_flip);
2349:   } else {
2350:     VecCopy(standard_rhs,pcbddc->original_rhs);
2351:   }
2352:   if (mat_ctx->g2g_p) {
2353:     /* interface pressure rhs */
2354:     VecScatterBegin(mat_ctx->g2g_p,fetidp_flux_rhs,pcbddc->original_rhs,INSERT_VALUES,SCATTER_REVERSE);
2355:     VecScatterEnd(mat_ctx->g2g_p,fetidp_flux_rhs,pcbddc->original_rhs,INSERT_VALUES,SCATTER_REVERSE);
2356:     VecScatterBegin(mat_ctx->g2g_p,standard_rhs,fetidp_flux_rhs,INSERT_VALUES,SCATTER_FORWARD);
2357:     VecScatterEnd(mat_ctx->g2g_p,standard_rhs,fetidp_flux_rhs,INSERT_VALUES,SCATTER_FORWARD);
2358:     if (!mat_ctx->rhs_flip) {
2359:       VecScale(fetidp_flux_rhs,-1.);
2360:     }
2361:   }
2362:   /*
2363:      change of basis for physical rhs if needed
2364:      It also changes the rhs in case of dirichlet boundaries
2365:   */
2366:   PCPreSolve_BDDC(mat_ctx->pc,NULL,pcbddc->original_rhs,NULL);
2367:   if (pcbddc->ChangeOfBasisMatrix) {
2368:     MatMultTranspose(pcbddc->ChangeOfBasisMatrix,pcbddc->original_rhs,pcbddc->work_change);
2369:     work = pcbddc->work_change;
2370:    } else {
2371:     work = pcbddc->original_rhs;
2372:   }
2373:   /* store vectors for computation of fetidp final solution */
2374:   VecScatterBegin(pcis->global_to_D,work,mat_ctx->temp_solution_D,INSERT_VALUES,SCATTER_FORWARD);
2375:   VecScatterEnd(pcis->global_to_D,work,mat_ctx->temp_solution_D,INSERT_VALUES,SCATTER_FORWARD);
2376:   /* scale rhs since it should be unassembled */
2377:   /* TODO use counter scaling? (also below) */
2378:   VecScatterBegin(pcis->global_to_B,work,mat_ctx->temp_solution_B,INSERT_VALUES,SCATTER_FORWARD);
2379:   VecScatterEnd(pcis->global_to_B,work,mat_ctx->temp_solution_B,INSERT_VALUES,SCATTER_FORWARD);
2380:   /* Apply partition of unity */
2381:   VecPointwiseMult(mat_ctx->temp_solution_B,pcis->D,mat_ctx->temp_solution_B);
2382:   /* PCBDDCScalingRestriction(mat_ctx->pc,work,mat_ctx->temp_solution_B); */
2383:   if (!pcbddc->switch_static) {
2384:     /* compute partially subassembled Schur complement right-hand side */
2385:     KSPSolve(pcbddc->ksp_D,mat_ctx->temp_solution_D,pcis->vec1_D);
2386:     /* Cannot propagate up error in KSPSolve() because there is no access to the PC */
2387:     MatMult(pcis->A_BI,pcis->vec1_D,pcis->vec1_B);
2388:     VecAXPY(mat_ctx->temp_solution_B,-1.0,pcis->vec1_B);
2389:     VecSet(work,0.0);
2390:     VecScatterBegin(pcis->global_to_B,mat_ctx->temp_solution_B,work,ADD_VALUES,SCATTER_REVERSE);
2391:     VecScatterEnd(pcis->global_to_B,mat_ctx->temp_solution_B,work,ADD_VALUES,SCATTER_REVERSE);
2392:     /* PCBDDCScalingRestriction(mat_ctx->pc,work,mat_ctx->temp_solution_B); */
2393:     VecScatterBegin(pcis->global_to_B,work,mat_ctx->temp_solution_B,INSERT_VALUES,SCATTER_FORWARD);
2394:     VecScatterEnd(pcis->global_to_B,work,mat_ctx->temp_solution_B,INSERT_VALUES,SCATTER_FORWARD);
2395:     VecPointwiseMult(mat_ctx->temp_solution_B,pcis->D,mat_ctx->temp_solution_B);
2396:   }
2397:   /* BDDC rhs */
2398:   VecCopy(mat_ctx->temp_solution_B,pcis->vec1_B);
2399:   if (pcbddc->switch_static) {
2400:     VecCopy(mat_ctx->temp_solution_D,pcis->vec1_D);
2401:   }
2402:   /* apply BDDC */
2403:   PetscArrayzero(pcbddc->benign_p0,pcbddc->benign_n);
2404:   PCBDDCApplyInterfacePreconditioner(mat_ctx->pc,PETSC_FALSE);
2405:   PetscArrayzero(pcbddc->benign_p0,pcbddc->benign_n);

2407:   /* Application of B_delta and assembling of rhs for fetidp fluxes */
2408:   MatMult(mat_ctx->B_delta,pcis->vec1_B,mat_ctx->lambda_local);
2409:   VecScatterBegin(mat_ctx->l2g_lambda,mat_ctx->lambda_local,fetidp_flux_rhs,ADD_VALUES,SCATTER_FORWARD);
2410:   VecScatterEnd(mat_ctx->l2g_lambda,mat_ctx->lambda_local,fetidp_flux_rhs,ADD_VALUES,SCATTER_FORWARD);
2411:   /* Add contribution to interface pressures */
2412:   if (mat_ctx->l2g_p) {
2413:     MatMult(mat_ctx->B_BB,pcis->vec1_B,mat_ctx->vP);
2414:     if (pcbddc->switch_static) {
2415:       MatMultAdd(mat_ctx->B_BI,pcis->vec1_D,mat_ctx->vP,mat_ctx->vP);
2416:     }
2417:     VecScatterBegin(mat_ctx->l2g_p,mat_ctx->vP,fetidp_flux_rhs,ADD_VALUES,SCATTER_FORWARD);
2418:     VecScatterEnd(mat_ctx->l2g_p,mat_ctx->vP,fetidp_flux_rhs,ADD_VALUES,SCATTER_FORWARD);
2419:   }
2420:   return(0);
2421: }

2455: static PetscErrorCode PCBDDCMatFETIDPGetSolution_BDDC(Mat fetidp_mat, Vec fetidp_flux_sol, Vec standard_sol)
2456: {
2457:   FETIDPMat_ctx  mat_ctx;
2458:   PC_IS*         pcis;
2459:   PC_BDDC*       pcbddc;
2461:   Vec            work;

2464:   MatShellGetContext(fetidp_mat,&mat_ctx);
2465:   pcis = (PC_IS*)mat_ctx->pc->data;
2466:   pcbddc = (PC_BDDC*)mat_ctx->pc->data;

2468:   /* apply B_delta^T */
2469:   VecSet(pcis->vec1_B,0.);
2470:   VecScatterBegin(mat_ctx->l2g_lambda,fetidp_flux_sol,mat_ctx->lambda_local,INSERT_VALUES,SCATTER_REVERSE);
2471:   VecScatterEnd(mat_ctx->l2g_lambda,fetidp_flux_sol,mat_ctx->lambda_local,INSERT_VALUES,SCATTER_REVERSE);
2472:   MatMultTranspose(mat_ctx->B_delta,mat_ctx->lambda_local,pcis->vec1_B);
2473:   if (mat_ctx->l2g_p) {
2474:     VecScatterBegin(mat_ctx->l2g_p,fetidp_flux_sol,mat_ctx->vP,INSERT_VALUES,SCATTER_REVERSE);
2475:     VecScatterEnd(mat_ctx->l2g_p,fetidp_flux_sol,mat_ctx->vP,INSERT_VALUES,SCATTER_REVERSE);
2476:     MatMultAdd(mat_ctx->Bt_BB,mat_ctx->vP,pcis->vec1_B,pcis->vec1_B);
2477:   }

2479:   /* compute rhs for BDDC application */
2480:   VecAYPX(pcis->vec1_B,-1.0,mat_ctx->temp_solution_B);
2481:   if (pcbddc->switch_static) {
2482:     VecCopy(mat_ctx->temp_solution_D,pcis->vec1_D);
2483:     if (mat_ctx->l2g_p) {
2484:       VecScale(mat_ctx->vP,-1.);
2485:       MatMultAdd(mat_ctx->Bt_BI,mat_ctx->vP,pcis->vec1_D,pcis->vec1_D);
2486:     }
2487:   }

2489:   /* apply BDDC */
2490:   PetscArrayzero(pcbddc->benign_p0,pcbddc->benign_n);
2491:   PCBDDCApplyInterfacePreconditioner(mat_ctx->pc,PETSC_FALSE);

2493:   /* put values into global vector */
2494:   if (pcbddc->ChangeOfBasisMatrix) work = pcbddc->work_change;
2495:   else work = standard_sol;
2496:   VecScatterBegin(pcis->global_to_B,pcis->vec1_B,work,INSERT_VALUES,SCATTER_REVERSE);
2497:   VecScatterEnd(pcis->global_to_B,pcis->vec1_B,work,INSERT_VALUES,SCATTER_REVERSE);
2498:   if (!pcbddc->switch_static) {
2499:     /* compute values into the interior if solved for the partially subassembled Schur complement */
2500:     MatMult(pcis->A_IB,pcis->vec1_B,pcis->vec1_D);
2501:     VecAYPX(pcis->vec1_D,-1.0,mat_ctx->temp_solution_D);
2502:     KSPSolve(pcbddc->ksp_D,pcis->vec1_D,pcis->vec1_D);
2503:     /* Cannot propagate up error in KSPSolve() because there is no access to the PC */
2504:   }

2506:   VecScatterBegin(pcis->global_to_D,pcis->vec1_D,work,INSERT_VALUES,SCATTER_REVERSE);
2507:   VecScatterEnd(pcis->global_to_D,pcis->vec1_D,work,INSERT_VALUES,SCATTER_REVERSE);
2508:   /* add p0 solution to final solution */
2509:   PCBDDCBenignGetOrSetP0(mat_ctx->pc,work,PETSC_FALSE);
2510:   if (pcbddc->ChangeOfBasisMatrix) {
2511:     MatMult(pcbddc->ChangeOfBasisMatrix,work,standard_sol);
2512:   }
2513:   PCPostSolve_BDDC(mat_ctx->pc,NULL,NULL,standard_sol);
2514:   if (mat_ctx->g2g_p) {
2515:     VecScatterBegin(mat_ctx->g2g_p,fetidp_flux_sol,standard_sol,INSERT_VALUES,SCATTER_REVERSE);
2516:     VecScatterEnd(mat_ctx->g2g_p,fetidp_flux_sol,standard_sol,INSERT_VALUES,SCATTER_REVERSE);
2517:   }
2518:   return(0);
2519: }
