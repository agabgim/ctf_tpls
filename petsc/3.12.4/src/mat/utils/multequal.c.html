457: /*@
458:    MatIsLinear - Check if a shell matrix A is a linear operator.

460:    Collective on Mat

462:    Input Parameters:
463: +  A - the shell matrix
464: -  n - number of random vectors to be tested

466:    Output Parameter:
467: .  flg - PETSC_TRUE if the shell matrix is linear; PETSC_FALSE otherwise.

469:    Level: intermediate
470: @*/
471: PetscErrorCode MatIsLinear(Mat A,PetscInt n,PetscBool  *flg)
472: {
474:   Vec            x,y,s1,s2;
475:   PetscRandom    rctx;
476:   PetscScalar    a;
477:   PetscInt       k;
478:   PetscReal      norm,normA;
479:   MPI_Comm       comm;
480:   PetscMPIInt    rank;

484:   PetscObjectGetComm((PetscObject)A,&comm);
485:   MPI_Comm_rank(comm,&rank);

487:   PetscRandomCreate(comm,&rctx);
488:   PetscRandomSetFromOptions(rctx);
489:   MatCreateVecs(A,&x,&s1);
490:   VecDuplicate(x,&y);
491:   VecDuplicate(s1,&s2);

493:   *flg = PETSC_TRUE;
494:   for (k=0; k<n; k++) {
495:     VecSetRandom(x,rctx);
496:     VecSetRandom(y,rctx);
497:     if (!rank) {
498:       PetscRandomGetValue(rctx,&a);
499:     }
500:     MPI_Bcast(&a, 1, MPIU_SCALAR, 0, comm);

502:     /* s2 = a*A*x + A*y */
503:     MatMult(A,y,s2); /* s2 = A*y */
504:     MatMult(A,x,s1); /* s1 = A*x */
505:     VecAXPY(s2,a,s1); /* s2 = a s1 + s2 */

507:     /* s1 = A * (a x + y) */
508:     VecAXPY(y,a,x); /* y = a x + y */
509:     MatMult(A,y,s1);
510:     VecNorm(s1,NORM_INFINITY,&normA);

512:     VecAXPY(s2,-1.0,s1); /* s2 = - s1 + s2 */
513:     VecNorm(s2,NORM_INFINITY,&norm);
514:     if (norm/normA > 100.*PETSC_MACHINE_EPSILON) {
515:       *flg = PETSC_FALSE;
516:       PetscInfo3(A,"Error: %D-th |A*(ax+y) - (a*A*x+A*y)|/|A(ax+y)| %g > tol %g\n",k,(double)norm/normA,100.*PETSC_MACHINE_EPSILON);
517:       break;
518:     }
519:   }
520:   PetscRandomDestroy(&rctx);
521:   VecDestroy(&x);
522:   VecDestroy(&y);
523:   VecDestroy(&s1);
524:   VecDestroy(&s2);
525:   return(0);
526: }
