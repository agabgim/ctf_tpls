285: static PetscErrorCode VecNorm_Nest(Vec xin,NormType type,PetscReal *z)
286: {
287:   Vec_Nest       *bx = (Vec_Nest*)xin->data;
288:   PetscInt       i,nr;
289:   PetscReal      z_i;
290:   PetscReal      _z;

294:   nr = bx->nb;
295:   _z = 0.0;

297:   if (type == NORM_2) {
298:     PetscScalar dot;
299:     VecDot(xin,xin,&dot);
300:     _z = PetscAbsScalar(PetscSqrtScalar(dot));
301:   } else if (type == NORM_1) {
302:     for (i=0; i<nr; i++) {
303:       VecNorm(bx->v[i],type,&z_i);
304:       _z = _z + z_i;
305:     }
306:   } else if (type == NORM_INFINITY) {
307:     for (i=0; i<nr; i++) {
308:       VecNorm(bx->v[i],type,&z_i);
309:       if (z_i > _z) _z = z_i;
310:     }
311:   }

313:   *z = _z;
314:   return(0);
315: }

416: static PetscErrorCode VecMax_Nest_Recursive(Vec x,PetscInt *cnt,PetscInt *p,PetscReal *max)
417: {
418:   Vec_Nest       *bx;
419:   PetscInt       i,nr;
420:   PetscBool      isnest;
421:   PetscInt       L;
422:   PetscInt       _entry_loc;
423:   PetscReal      _entry_val;

427:   PetscObjectTypeCompare((PetscObject)x,VECNEST,&isnest);
428:   if (!isnest) {
429:     /* Not nest */
430:     VecMax(x,&_entry_loc,&_entry_val);
431:     if (_entry_val > *max) {
432:       *max = _entry_val;
433:       if(p) *p = _entry_loc + *cnt;
434:     }
435:     VecGetSize(x,&L);
436:     *cnt = *cnt + L;
437:     return(0);
438:   }

440:   /* Otherwise we have a nest */
441:   bx = (Vec_Nest*)x->data;
442:   nr = bx->nb;

444:   /* now descend recursively */
445:   for (i=0; i<nr; i++) {
446:     VecMax_Nest_Recursive(bx->v[i],cnt,p,max);
447:   }
448:   return(0);
449: }

638: static PetscErrorCode VecGetArray_Nest(Vec X,PetscScalar **x)
639: {
640:   Vec_Nest       *bx = (Vec_Nest*)X->data;
641:   PetscInt       i,m,rstart,rend;

645:   VecGetOwnershipRange(X,&rstart,&rend);
646:   VecGetLocalSize(X,&m);

648:   for (i=0; i<bx->nb; i++) {

653:     const PetscScalar *y;
654:     VecGetLocalSize(subvec,&sm);
655:     VecGetArrayRead(subvec,&y);
656:     ISGetIndices(isy,&ixy);
657:     for (j=0; j<sm; j++) {
658:       PetscInt ix = ixy[j];
659:       if (ix < rstart || rend <= ix) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_SUP,"No support for getting array from nonlocal subvec");
660:       (*x)[ix-rstart] = y[j];
661:     }
662:     ISRestoreIndices(isy,&ixy);
663:     VecRestoreArrayRead(subvec,&y);
664:   }
665:   return(0);
666: }

872: static PetscErrorCode  VecNestSetSubVec_Private(Vec X,PetscInt idxm,Vec x)
873: {
874:   Vec_Nest       *bx = (Vec_Nest*)X->data;
875:   PetscInt       i,offset=0,n=0,bs;
876:   IS             is;
878:   PetscBool      issame = PETSC_FALSE;
879:   PetscInt       N=0;

881:   /* check if idxm < bx->nb */
882:   if (idxm >= bx->nb) SETERRQ2(PETSC_COMM_SELF,PETSC_ERR_ARG_OUTOFRANGE,"Out of range index value %D maximum %D",idxm,bx->nb);

885:   VecDestroy(&bx->v[idxm]);       /* destroy the existing vector */
886:   VecDuplicate(x,&bx->v[idxm]);   /* duplicate the layout of given vector */
887:   VecCopy(x,bx->v[idxm]);         /* copy the contents of the given vector */

889:   /* check if we need to update the IS for the block */
890:   offset = X->map->rstart;
891:   for (i=0; i<idxm; i++) {
892:     n=0;
893:     VecGetLocalSize(bx->v[i],&n);
894:     offset += n;
895:   }

897:   /* get the local size and block size */
898:   VecGetLocalSize(x,&n);
899:   VecGetBlockSize(x,&bs);

901:   /* create the new IS */
902:   ISCreateStride(PetscObjectComm((PetscObject)x),n,offset,1,&is);
903:   ISSetBlockSize(is,bs);

905:   /* check if they are equal */
906:   ISEqual(is,bx->is[idxm],&issame);

908:   if (!issame) {
909:     /* The IS of given vector has a different layout compared to the existing block vector.
910:      Destroy the existing reference and update the IS. */
911:     ISDestroy(&bx->is[idxm]);
912:     ISDuplicate(is,&bx->is[idxm]);
913:     ISCopy(is,bx->is[idxm]);

915:     offset += n;
916:     /* Since the current IS[idxm] changed, we need to update all the subsequent IS */
917:     for (i=idxm+1; i<bx->nb; i++) {
918:       /* get the local size and block size */
919:       VecGetLocalSize(bx->v[i],&n);
920:       VecGetBlockSize(bx->v[i],&bs);

922:       /* destroy the old and create the new IS */
923:       ISDestroy(&bx->is[i]);
924:       ISCreateStride(((PetscObject)bx->v[i])->comm,n,offset,1,&bx->is[i]);
925:       ISSetBlockSize(bx->is[i],bs);

927:       offset += n;
928:     }

930:     n=0;
931:     VecSize_Nest_Recursive(X,PETSC_TRUE,&N);
932:     VecSize_Nest_Recursive(X,PETSC_FALSE,&n);
933:     PetscLayoutSetSize(X->map,N);
934:     PetscLayoutSetLocalSize(X->map,n);
935:   }

937:   ISDestroy(&is);
938:   return(0);
939: }

1053: static PetscErrorCode VecSetUp_Nest_Private(Vec V,PetscInt nb,Vec x[])
1054: {
1055:   Vec_Nest       *ctx = (Vec_Nest*)V->data;
1056:   PetscInt       i;

1060:   if (ctx->setup_called) return(0);

1062:   ctx->nb = nb;
1063:   if (ctx->nb < 0) SETERRQ(PetscObjectComm((PetscObject)V),PETSC_ERR_ARG_WRONG,"Cannot create VECNEST with < 0 blocks.");

1065:   /* Create space */
1066:   PetscMalloc1(ctx->nb,&ctx->v);
1067:   for (i=0; i<ctx->nb; i++) {
1068:     ctx->v[i] = x[i];
1069:     PetscObjectReference((PetscObject)x[i]);
1070:     /* Do not allocate memory for internal sub blocks */
1071:   }

1073:   PetscMalloc1(ctx->nb,&ctx->is);

1075:   ctx->setup_called = PETSC_TRUE;
1076:   return(0);
1077: }

1079: static PetscErrorCode VecSetUp_NestIS_Private(Vec V,PetscInt nb,IS is[])
1080: {
1081:   Vec_Nest       *ctx = (Vec_Nest*)V->data;
1082:   PetscInt       i,offset,m,n,M,N;

1086:   if (is) {                     /* Do some consistency checks and reference the is */
1087:     offset = V->map->rstart;
1088:     for (i=0; i<ctx->nb; i++) {
1089:       ISGetSize(is[i],&M);
1090:       VecGetSize(ctx->v[i],&N);
1091:       if (M != N) SETERRQ3(PetscObjectComm((PetscObject)V),PETSC_ERR_ARG_INCOMP,"In slot %D, IS of size %D is not compatible with Vec of size %D",i,M,N);
1092:       ISGetLocalSize(is[i],&m);
1093:       VecGetLocalSize(ctx->v[i],&n);
1094:       if (m != n) SETERRQ3(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"In slot %D, IS of local size %D is not compatible with Vec of local size %D",i,m,n);
1095: #if defined(PETSC_USE_DEBUG)
1096:       {                         /* This test can be expensive */
1097:         PetscInt  start;
1098:         PetscBool contiguous;
1099:         ISContiguousLocal(is[i],offset,offset+n,&start,&contiguous);
1100:         if (!contiguous) SETERRQ1(PetscObjectComm((PetscObject)V),PETSC_ERR_SUP,"Index set %D is not contiguous with layout of matching vector",i);
1101:         if (start != 0) SETERRQ1(PetscObjectComm((PetscObject)V),PETSC_ERR_SUP,"Index set %D introduces overlap or a hole",i);
1102:       }
1103: #endif
1104:       PetscObjectReference((PetscObject)is[i]);
1105:       ctx->is[i] = is[i];
1106:       offset += n;
1107:     }
1108:   } else {                      /* Create a contiguous ISStride for each entry */
1109:     offset = V->map->rstart;
1110:     for (i=0; i<ctx->nb; i++) {
1111:       PetscInt bs;
1112:       VecGetLocalSize(ctx->v[i],&n);
1113:       VecGetBlockSize(ctx->v[i],&bs);
1114:       ISCreateStride(((PetscObject)ctx->v[i])->comm,n,offset,1,&ctx->is[i]);
1115:       ISSetBlockSize(ctx->is[i],bs);
1116:       offset += n;
1117:     }
1118:   }
1119:   return(0);
1120: }

1122: /*@C
1123:    VecCreateNest - Creates a new vector containing several nested subvectors, each stored separately

1125:    Collective on Vec

1127:    Input Parameter:
1128: +  comm - Communicator for the new Vec
1129: .  nb - number of nested blocks
1130: .  is - array of nb index sets describing each nested block, or NULL to pack subvectors contiguously
1131: -  x - array of nb sub-vectors

1133:    Output Parameter:
1134: .  Y - new vector

1136:    Level: advanced

1138: .seealso: VecCreate(), MatCreateNest(), DMSetVecType(), VECNEST
1139: @*/
1140: PetscErrorCode  VecCreateNest(MPI_Comm comm,PetscInt nb,IS is[],Vec x[],Vec *Y)
1141: {
1142:   Vec            V;
1143:   Vec_Nest       *s;
1144:   PetscInt       n,N;

1148:   VecCreate(comm,&V);
1149:   PetscObjectChangeTypeName((PetscObject)V,VECNEST);

1151:   /* allocate and set pointer for implememtation data */

1153:   V->data          = (void*)s;
1154:   s->setup_called  = PETSC_FALSE;
1155:   s->nb            = -1;
1156:   s->v             = NULL;

1158:   VecSetUp_Nest_Private(V,nb,x);

1160:   n = N = 0;
1161:   VecSize_Nest_Recursive(V,PETSC_TRUE,&N);
1162:   VecSize_Nest_Recursive(V,PETSC_FALSE,&n);

1164:   PetscLayoutSetLocalSize(V->map,n);
1165:   PetscLayoutSetBlockSize(V->map,1);
1166:   PetscLayoutSetUp(V->map);

1168:   VecSetUp_NestIS_Private(V,nb,is);

1170:   VecNestSetOps_Private(V->ops);
1171:   V->petscnative = PETSC_FALSE;


1174:   /* expose block api's */
1175:   PetscObjectComposeFunction((PetscObject)V,"VecNestGetSubVec_C",VecNestGetSubVec_Nest);

1179:   PetscObjectComposeFunction((PetscObject)V,"VecNestGetSize_C",VecNestGetSize_Nest);

1181:   *Y = V;
1182:   return(0);
1183: }
