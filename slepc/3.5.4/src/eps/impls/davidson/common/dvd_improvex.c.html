 87: PetscErrorCode dvd_improvex_jd(dvdDashboard *d,dvdBlackboard *b,KSP ksp,PetscInt max_bs,PetscInt cX_impr,PetscBool dynamic)
 88: {
 89:   PetscErrorCode  ierr;
 90:   dvdImprovex_jd  *data;
 91:   PetscBool       useGD;
 92:   PC              pc;
 93:   PetscInt        size_P;


 97:   /* Setting configuration constrains */
 98:   PetscObjectTypeCompare((PetscObject)ksp,KSPPREONLY,&useGD);

100:   /* If the arithmetic is real and the problem is not Hermitian, then
101:      the block size is incremented in one */
102: #if !defined(PETSC_USE_COMPLEX)
103:   if (!DVD_IS(d->sEP,DVD_EP_HERMITIAN)) {
104:     max_bs++;
105:     b->max_size_P = PetscMax(b->max_size_P,2);
106:   } else
107: #endif
108:   {
109:     b->max_size_P = PetscMax(b->max_size_P,1);
110:   }
111:   b->max_size_X = PetscMax(b->max_size_X,max_bs);
112:   size_P = b->max_size_P+cX_impr;

114:   /* Setup the preconditioner */
115:   if (ksp) {
116:     KSPGetPC(ksp,&pc);
117:     dvd_static_precond_PC(d,b,pc);
118:   } else {
119:     dvd_static_precond_PC(d,b,0);
120:   }

122:   /* Setup the step */
123:   if (b->state >= DVD_STATE_CONF) {
124:     PetscMalloc(sizeof(dvdImprovex_jd),&data);
125:     PetscLogObjectMemory((PetscObject)d->eps,sizeof(dvdImprovex_jd));
126:     data->dynamic = dynamic;
127:     d->max_cX_in_impr = cX_impr;
128:     PetscMalloc1(size_P*size_P,&data->XKZ);
129:     PetscMalloc1(size_P*size_P,&data->iXKZ);
130:     PetscMalloc1(size_P,&data->iXKZPivots);
131:     data->ldXKZ = size_P;
132:     data->size_X = b->max_size_X;
133:     d->improveX_data = data;
134:     data->ksp = useGD?NULL:ksp;
135:     data->d = d;
136:     d->improveX = dvd_improvex_jd_gen;
137: #if !defined(PETSC_USE_COMPLEX)
138:     if (!DVD_IS(d->sEP,DVD_EP_HERMITIAN)) {
139:       data->ksp_max_size = 2;
140:     } else
141: #endif
142:     {
143:       data->ksp_max_size = 1;
144:     }
145:     /* Create various vector basis */
146:     BVDuplicateResize(d->eps->V,size_P,&data->KZ);
147:     BVSetMatrix(data->KZ,NULL,PETSC_FALSE);
148:     BVDuplicate(data->KZ,&data->U);

150:     EPSDavidsonFLAdd(&d->startList,dvd_improvex_jd_start);
151:     EPSDavidsonFLAdd(&d->endList,dvd_improvex_jd_end);
152:     EPSDavidsonFLAdd(&d->destroyList,dvd_improvex_jd_d);
153:   }
154:   return(0);
155: }

159: static PetscErrorCode dvd_improvex_jd_start(dvdDashboard *d)
160: {
161:   PetscErrorCode  ierr;
162:   dvdImprovex_jd  *data = (dvdImprovex_jd*)d->improveX_data;
163:   PetscInt        rA, cA, rlA, clA;
164:   Mat             A;
165:   PetscBool       t;
166:   PC              pc;
167:   Vec             v0[2];

170:   data->size_cX = data->old_size_X = 0;
171:   data->lastTol = data->dynamic?0.5:0.0;

173:   /* Setup the ksp */
174:   if (data->ksp) {
175:     /* Create the reference vector */
176:     BVGetColumn(d->eps->V,0,&v0[0]);
177:     v0[1] = v0[0];
178:     VecCreateCompWithVecs(v0,data->ksp_max_size,NULL,&data->friends);
179:     BVRestoreColumn(d->eps->V,0,&v0[0]);
180:     PetscLogObjectParent((PetscObject)d->eps,(PetscObject)data->friends);

182:     /* Save the current pc and set a PCNONE */
183:     KSPGetPC(data->ksp, &data->old_pc);
184:     PetscObjectTypeCompare((PetscObject)data->old_pc,PCNONE,&t);
185:     data->lastTol = 0.5;
186:     if (t) {
187:       data->old_pc = 0;
188:     } else {
189:       PetscObjectReference((PetscObject)data->old_pc);
190:       PCCreate(PetscObjectComm((PetscObject)d->eps),&pc);
191:       PCSetType(pc,PCSHELL);
192:       PCSetOperators(pc,d->A,d->A);
193:       PCSetReusePreconditioner(pc,PETSC_TRUE);
194:       PCShellSetApply(pc,PCApply_dvd);
195:       PCShellSetApplyBA(pc,PCApplyBA_dvd);
196:       PCShellSetApplyTranspose(pc,PCApplyTranspose_dvd);
197:       KSPSetPC(data->ksp,pc);
198:       PCDestroy(&pc);
199:     }

201:     /* Create the (I-v*u')*K*(A-s*B) matrix */
202:     MatGetSize(d->A, &rA, &cA);
203:     MatGetLocalSize(d->A, &rlA, &clA);
204:     MatCreateShell(PetscObjectComm((PetscObject)d->A), rlA*data->ksp_max_size,
205:                           clA*data->ksp_max_size, rA*data->ksp_max_size,
206:                           cA*data->ksp_max_size, data, &A);
207:     MatShellSetOperation(A, MATOP_MULT,(void(*)(void))MatMult_dvd_jd);
208:     MatShellSetOperation(A, MATOP_MULT_TRANSPOSE,(void(*)(void))MatMultTranspose_dvd_jd);
209:     MatShellSetOperation(A, MATOP_GET_VECS,(void(*)(void))MatGetVecs_dvd_jd);

211:     /* Try to avoid KSPReset */
212:     KSPGetOperatorsSet(data->ksp,&t,NULL);
213:     if (t) {
214:       Mat M;
215:       PetscInt rM;
216:       KSPGetOperators(data->ksp,&M,NULL);
217:       MatGetSize(M,&rM,NULL);
218:       if (rM != rA*data->ksp_max_size) { KSPReset(data->ksp); }
219:     }
220:     KSPSetOperators(data->ksp,A,A);
221:     KSPSetReusePreconditioner(data->ksp,PETSC_TRUE);
222:     KSPSetUp(data->ksp);
223:     MatDestroy(&A);
224:   } else {
225:     data->old_pc = 0;
226:     data->friends = NULL;
227:   }
228:   BVSetActiveColumns(data->KZ,0,0);
229:   BVSetActiveColumns(data->U,0,0);
230:   return(0);
231: }

271: static PetscErrorCode dvd_improvex_jd_gen(dvdDashboard *d,PetscInt r_s,PetscInt r_e,PetscInt *size_D)
272: {
273:   dvdImprovex_jd  *data = (dvdImprovex_jd*)d->improveX_data;
274:   PetscErrorCode  ierr;
275:   PetscInt        i,j,n,maxits,maxits0,lits,s,ld,k,max_size_D,lV,kV;
276:   PetscScalar     *pX,*pY;
277:   PetscReal       tol,tol0;
278:   Vec             *kr,kr_comp,D_comp,D[2],kr0[2];
279:   PetscBool       odd_situation = PETSC_FALSE;

282:   BVGetActiveColumns(d->eps->V,&lV,&kV);
283:   max_size_D = d->eps->ncv-kV;
284:   /* Quick exit */
285:   if ((max_size_D == 0) || r_e-r_s <= 0) {
286:    *size_D = 0;
287:     return(0);
288:   }

290:   n = PetscMin(PetscMin(data->size_X, max_size_D), r_e-r_s);
291:   if (n == 0) SETERRQ(PETSC_COMM_SELF,1, "n == 0");
292:   if (data->size_X < r_e-r_s) SETERRQ(PETSC_COMM_SELF,1, "size_X < r_e-r_s");

294:   DSGetLeadingDimension(d->eps->ds,&ld);

296:   /* Restart lastTol if a new pair converged */
297:   if (data->dynamic && data->size_cX < lV)
298:     data->lastTol = 0.5;

300:   for (i=0,s=0;i<n;i+=s) {
301:     /* If the selected eigenvalue is complex, but the arithmetic is real... */
302: #if !defined(PETSC_USE_COMPLEX)
303:     if (d->eigi[i] != 0.0) {
304:       if (i+2 <= max_size_D) s=2;
305:       else break;
306:     } else
307: #endif
308:       s=1;

310:     data->r_s = r_s+i; data->r_e = r_s+i+s;
311:     SlepcVecPoolGetVecs(d->auxV,s,&kr);

313:     /* Compute theta, maximum iterations and tolerance */
314:     maxits = 0; tol = 1;
315:     for (j=0;j<s;j++) {
316:       d->improvex_jd_lit(d,r_s+i+j,&data->theta[2*j],&data->thetai[j],&maxits0,&tol0);
317:       maxits+= maxits0; tol*= tol0;
318:     }
319:     maxits/= s; tol = data->dynamic?data->lastTol:PetscExpReal(PetscLogReal(tol)/s);

321:     /* Compute u, v and kr */
322:     k = r_s+i;
323:     DSVectors(d->eps->ds,DS_MAT_X,&k,NULL);
324:     DSNormalize(d->eps->ds,DS_MAT_X,r_s+i);
325:     k = r_s+i;
326:     DSVectors(d->eps->ds,DS_MAT_Y,&k,NULL);
327:     DSNormalize(d->eps->ds,DS_MAT_Y,r_s+i);
328:     DSGetArray(d->eps->ds,DS_MAT_X,&pX);
329:     DSGetArray(d->eps->ds,DS_MAT_Y,&pY);
330:     dvd_improvex_jd_proj_cuv(d,r_s+i,r_s+i+s,kr,data->theta,data->thetai,pX,pY,ld);
331:     DSRestoreArray(d->eps->ds,DS_MAT_X,&pX);
332:     DSRestoreArray(d->eps->ds,DS_MAT_Y,&pY);

334:     /* Check if the first eigenpairs are converged */
335:     if (i == 0) {
336:       d->preTestConv(d,0,s,s,&d->npreconv);
337:       if (d->npreconv > 0) break;
338:     }

340:     /* Test the odd situation of solving Ax=b with A=I */
341: #if !defined(PETSC_USE_COMPLEX)
342:     odd_situation = (data->ksp && data->theta[0] == 1. && data->theta[1] == 0. && data->thetai[0] == 0. && d->B == NULL)?PETSC_TRUE:PETSC_FALSE;
343: #else
344:     odd_situation = (data->ksp && data->theta[0] == 1. && data->theta[1] == 0. && d->B == NULL)?PETSC_TRUE:PETSC_FALSE;
345: #endif
346:     /* If JD */
347:     if (data->ksp && !odd_situation) {
348:       /* kr <- -kr */
349:       for (j=0;j<s;j++) {
350:         VecScale(kr[j],-1.0);
351:       }

353:       /* Compose kr and D */
354:       kr0[0] = kr[0];
355:       kr0[1] = (s==2 ? kr[1] : NULL);
356:       VecCreateCompWithVecs(kr0,data->ksp_max_size,data->friends,&kr_comp);
357:       BVGetColumn(d->eps->V,kV+r_s+i,&D[0]);
358:       if (s==2) { BVGetColumn(d->eps->V,kV+r_s+i+1,&D[1]); }
359:       else D[1] = NULL;
360:       VecCreateCompWithVecs(D,data->ksp_max_size,data->friends,&D_comp);
361:       VecCompSetSubVecs(data->friends,s,NULL);

363:       /* Solve the correction equation */
364:       KSPSetTolerances(data->ksp,tol,PETSC_DEFAULT,PETSC_DEFAULT,maxits);
365:       KSPSolve(data->ksp,kr_comp,D_comp);
366:       KSPGetIterationNumber(data->ksp,&lits);

368:       /* Destroy the composed ks and D */
369:       VecDestroy(&kr_comp);
370:       VecDestroy(&D_comp);
371:       BVRestoreColumn(d->eps->V,kV+r_s+i,&D[0]);
372:       if (s==2) { BVRestoreColumn(d->eps->V,kV+r_s+i+1,&D[1]); }

374:     /* If GD */
375:     } else {
376:       BVGetColumn(d->eps->V,kV+r_s+i,&D[0]);
377:       if (s==2) { BVGetColumn(d->eps->V,kV+r_s+i+1,&D[1]); }
378:       for (j=0;j<s;j++) {
379:         d->improvex_precond(d,r_s+i+j,kr[j],D[j]);
380:       }
381:       dvd_improvex_apply_proj(d,D,s);
382:       BVRestoreColumn(d->eps->V,kV+r_s+i,&D[0]);
383:       if (s==2) { BVRestoreColumn(d->eps->V,kV+r_s+i+1,&D[1]); }
384:     }
385:     /* Prevent that short vectors are discarded in the orthogonalization */
386:     if (i == 0 && d->eps->errest[d->nconv+r_s] > PETSC_MACHINE_EPSILON && d->eps->errest[d->nconv+r_s] < PETSC_MAX_REAL) {
387:       for (j=0;j<s;j++) {
388:         BVScaleColumn(d->eps->V,kV+r_s+i+j,1.0/d->eps->errest[d->nconv+r_s]);
389:       }
390:     }
391:     SlepcVecPoolRestoreVecs(d->auxV,s,&kr);
392:   }
393:   *size_D = i;
394:   if (data->dynamic) data->lastTol = PetscMax(data->lastTol/2.0,PETSC_MACHINE_EPSILON*10.0);

396:   return(0);
397: }

403: PETSC_STATIC_INLINE PetscErrorCode dvd_aux_matmult(dvdImprovex_jd *data,const Vec *x,const Vec *y)
404: {
405:   PetscErrorCode  ierr;
406:   PetscInt        n,i;
407:   const Vec       *Bx;
408:   Vec             *auxV;

411:   n = data->r_e - data->r_s;
412:   for (i=0;i<n;i++) {
413:     MatMult(data->d->A,x[i],y[i]);
414:   }

416:   SlepcVecPoolGetVecs(data->d->auxV,2,&auxV);
417:   for (i=0;i<n;i++) {
418: #if !defined(PETSC_USE_COMPLEX)
419:     if (data->d->eigi[data->r_s+i] != 0.0) {
420:       if (data->d->B) {
421:         MatMult(data->d->B,x[i],auxV[0]);
422:         MatMult(data->d->B,x[i+1],auxV[1]);
423:         Bx = auxV;
424:       } else Bx = &x[i];

426:       /* y_i   <- [ t_2i+1*A*x_i   - t_2i*Bx_i + ti_i*Bx_i+1;
427:          y_i+1      t_2i+1*A*x_i+1 - ti_i*Bx_i - t_2i*Bx_i+1  ] */
428:       VecAXPBYPCZ(y[i],-data->theta[2*i],data->thetai[i],data->theta[2*i+1],Bx[0],Bx[1]);
429:       VecAXPBYPCZ(y[i+1],-data->thetai[i],-data->theta[2*i],data->theta[2*i+1],Bx[0],Bx[1]);
430:       i++;
431:     } else
432: #endif
433:     {
434:       if (data->d->B) {
435:         MatMult(data->d->B,x[i],auxV[0]);
436:         Bx = auxV;
437:       } else Bx = &x[i];
438:       VecAXPBY(y[i],-data->theta[i*2],data->theta[i*2+1],Bx[0]);
439:     }
440:   }
441:   SlepcVecPoolRestoreVecs(data->d->auxV,2,&auxV);
442:   return(0);
443: }

445: /* y <- theta[1]'*A'*x - theta[0]'*B'*x */
448: PETSC_STATIC_INLINE PetscErrorCode dvd_aux_matmulttrans(dvdImprovex_jd *data,const Vec *x,const Vec *y)
449: {
450:   PetscErrorCode  ierr;
451:   PetscInt        n,i;
452:   const Vec       *Bx;
453:   Vec             *auxV;

456:   n = data->r_e - data->r_s;
457:   for (i=0;i<n;i++) {
458:     MatMultTranspose(data->d->A,x[i],y[i]);
459:   }

461:   SlepcVecPoolGetVecs(data->d->auxV,2,&auxV);
462:   for (i=0;i<n;i++) {
463: #if !defined(PETSC_USE_COMPLEX)
464:     if (data->d->eigi[data->r_s+i] != 0.0) {
465:       if (data->d->B) {
466:         MatMultTranspose(data->d->B,x[i],auxV[0]);
467:         MatMultTranspose(data->d->B,x[i+1],auxV[1]);
468:         Bx = auxV;
469:       } else Bx = &x[i];

471:       /* y_i   <- [ t_2i+1*A*x_i   - t_2i*Bx_i - ti_i*Bx_i+1;
472:          y_i+1      t_2i+1*A*x_i+1 + ti_i*Bx_i - t_2i*Bx_i+1  ] */
473:       VecAXPBYPCZ(y[i],-data->theta[2*i],-data->thetai[i],data->theta[2*i+1],Bx[0],Bx[1]);
474:       VecAXPBYPCZ(y[i+1],data->thetai[i],-data->theta[2*i],data->theta[2*i+1],Bx[0],Bx[1]);
475:       i++;
476:     } else
477: #endif
478:     {
479:       if (data->d->B) {

488: }

492: static PetscErrorCode PCApplyBA_dvd(PC pc,PCSide side,Vec in,Vec out,Vec w)
493: {
494:   PetscErrorCode  ierr;
495:   dvdImprovex_jd  *data;
496:   PetscInt        n,i;
497:   const Vec       *inx,*outx,*wx;
498:   Vec             *auxV;
499:   Mat             A;

502:   PCGetOperators(pc,&A,NULL);
503:   MatShellGetContext(A,(void**)&data);
504:   VecCompGetSubVecs(in,NULL,&inx);
505:   VecCompGetSubVecs(out,NULL,&outx);
506:   VecCompGetSubVecs(w,NULL,&wx);
507:   n = data->r_e - data->r_s;
508:   SlepcVecPoolGetVecs(data->d->auxV,n,&auxV);
509:   switch (side) {
510:   case PC_LEFT:
511:     /* aux <- theta[1]A*in - theta[0]*B*in */
512:     dvd_aux_matmult(data,inx,auxV);

514:     /* out <- K * aux */
515:     for (i=0;i<n;i++) {
516:       data->d->improvex_precond(data->d,data->r_s+i,auxV[i],outx[i]);
517:     }
518:     break;
519:   case PC_RIGHT:
520:     /* aux <- K * in */
521:     for (i=0;i<n;i++) {
522:       data->d->improvex_precond(data->d,data->r_s+i,inx[i],auxV[i]);
523:     }

525:     /* out <- theta[1]A*auxV - theta[0]*B*auxV */
526:     dvd_aux_matmult(data,auxV,outx);
527:     break;
528:   case PC_SYMMETRIC:
529:     /* aux <- K^{1/2} * in */
530:     for (i=0;i<n;i++) {
531:       PCApplySymmetricRight(data->old_pc,inx[i],auxV[i]);
532:     }

534:     /* wx <- theta[1]A*auxV - theta[0]*B*auxV */
535:     dvd_aux_matmult(data,auxV,wx);

537:     /* aux <- K^{1/2} * in */
538:     for (i=0;i<n;i++) {
539:       PCApplySymmetricLeft(data->old_pc,wx[i],outx[i]);
540:     }
541:     break;
542:   default:
543:     SETERRQ(PETSC_COMM_SELF,1, "Unsupported KSP side");
544:   }
545:   /* out <- out - v*(u'*out) */
546:   dvd_improvex_apply_proj(data->d,(Vec*)outx,n);
547:   SlepcVecPoolRestoreVecs(data->d->auxV,n,&auxV);
548:   return(0);
549: }

715: PetscErrorCode dvd_improvex_jd_proj_uv(dvdDashboard *d,dvdBlackboard *b,ProjType_t p)
716: {
718:   /* Setup the step */
719:   if (b->state >= DVD_STATE_CONF) {
720:     switch (p) {
721:     case DVD_PROJ_KXX:
722:       d->improvex_jd_proj_uv = dvd_improvex_jd_proj_uv_KXX; break;
723:     case DVD_PROJ_KZX:
724:       d->improvex_jd_proj_uv = dvd_improvex_jd_proj_uv_KZX; break;
725:     }
726:   }
727:   return(0);
728: }

732: /*
733:   Compute: u <- X, v <- K*(theta[0]*A+theta[1]*B)*X,
734:   kr <- K^{-1}*(A-eig*B)*X, being X <- V*pX[i_s..i_e-1], Y <- W*pY[i_s..i_e-1]
735:   where
736:   pX,pY, the right and left eigenvectors of the projected system
737:   ld, the leading dimension of pX and pY
738: */
739: static PetscErrorCode dvd_improvex_jd_proj_cuv(dvdDashboard *d,PetscInt i_s,PetscInt i_e,Vec *kr,PetscScalar *theta,PetscScalar *thetai,PetscScalar *pX,PetscScalar *pY,PetscInt ld)
740: {
741: #if defined(PETSC_MISSING_LAPACK_GETRF)
743:   SETERRQ(PETSC_COMM_SELF,PETSC_ERR_SUP,"GETRF - Lapack routine is unavailable");
744: #else
745:   PetscErrorCode    ierr;
746:   PetscInt          n=i_e-i_s,size_KZ,V_new,rm,i,lv,kv,lKZ,kKZ;
747:   dvdImprovex_jd    *data = (dvdImprovex_jd*)d->improveX_data;
748:   Vec               u[2],v[2];
749:   PetscBLASInt      s, ldXKZ, info;

752:   /* Check consistency */
753:   BVGetActiveColumns(d->eps->V,&lv,&kv);
754:   V_new = lv - data->size_cX;
755:   if (V_new > data->old_size_X) SETERRQ(PETSC_COMM_SELF,1, "Consistency broken");
756:   data->old_size_X = n;
757:   data->size_cX = lv;

759:   /* KZ <- KZ(rm:rm+max_cX-1) */
760:   BVGetActiveColumns(data->KZ,&lKZ,&kKZ);
761:   rm = PetscMax(V_new+lKZ-d->max_cX_in_impr,0);
762:   if (rm > 0) for (i=0; i<lKZ; i++) {
763:       BVCopyColumn(data->KZ,i+rm,i);
764:       BVCopyColumn(data->U,i+rm,i);
765:   }

767:   /* XKZ <- XKZ(rm:rm+max_cX-1,rm:rm+max_cX-1) */
768:   if (rm > 0) for (i=0; i<lKZ; i++) {
769:     PetscMemcpy(&data->XKZ[i*data->ldXKZ+i],&data->XKZ[(i+rm)*data->ldXKZ+i+rm],sizeof(PetscScalar)*lKZ);
770:   }
771:   lKZ = PetscMin(d->max_cX_in_impr,lKZ+V_new);
772:   BVSetActiveColumns(data->KZ,lKZ,lKZ+n);
773:   BVSetActiveColumns(data->U,lKZ,lKZ+n);

775:   /* Compute X, KZ and KR */
776:   BVGetColumn(data->U,lKZ,u);
777:   if (n>1) {BVGetColumn(data->U,lKZ+1,&u[1]);}
778:   BVGetColumn(data->KZ,lKZ,v);
779:   if (n>1) {BVGetColumn(data->KZ,lKZ+1,&v[1]);}
780:   d->improvex_jd_proj_uv(d,i_s,i_e,u,v,kr,theta,thetai,pX,pY,ld);
781:   BVRestoreColumn(data->U,lKZ,u);
782:   if (n>1) {BVRestoreColumn(data->U,lKZ+1,&u[1]);}
783:   BVRestoreColumn(data->KZ,lKZ,v);
784:   if (n>1) {BVRestoreColumn(data->KZ,lKZ+1,&v[1]);}

786:   /* XKZ <- U'*KZ */
787:   BVMultS(data->KZ,data->U,data->XKZ,data->ldXKZ);

789:   /* iXKZ <- inv(XKZ) */
790:   size_KZ = lKZ+n;
791:   PetscBLASIntCast(lKZ+n,&s);
792:   data->ldiXKZ = data->size_iXKZ = size_KZ;
793:   for (i=0;i<size_KZ;i++) {
794:     PetscMemcpy(&data->iXKZ[data->ldiXKZ*i],&data->XKZ[data->ldXKZ*i],sizeof(PetscScalar)*size_KZ);
795:   }
796:   PetscBLASIntCast(data->ldiXKZ,&ldXKZ);
797:   PetscFPTrapPush(PETSC_FP_TRAP_OFF);
798:   PetscStackCallBLAS("LAPACKgetrf",LAPACKgetrf_(&s, &s, data->iXKZ, &ldXKZ, data->iXKZPivots, &info));
799:   PetscFPTrapPop();
800:   if (info) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_LIB, "Error in Lapack XGETRF %d", info);
801:   return(0);
802: #endif
803: }

807: PETSC_STATIC_INLINE PetscErrorCode dvd_complex_rayleigh_quotient(Vec ur,Vec ui,Vec Axr,Vec Axi,Vec Bxr,Vec Bxi,PetscScalar *eigr,PetscScalar *eigi)
808: { 
809:   PetscErrorCode  ierr;
810:   PetscScalar     rAr,iAr,rAi,iAi,rBr,iBr,rBi,iBi,b0,b2,b4,b6,b7;

813:   /* eigr = [(rAr+iAi)*(rBr+iBi) + (rAi-iAr)*(rBi-iBr)]/k
814:      eigi = [(rAi-iAr)*(rBr+iBi) - (rAr+iAi)*(rBi-iBr)]/k
815:      k    =  (rBr+iBi)*(rBr+iBi) + (rBi-iBr)*(rBi-iBr)    */
816:   VecDotBegin(Axr,ur,&rAr); /* r*A*r */
817:   VecDotBegin(Axr,ui,&iAr); /* i*A*r */
818:   VecDotBegin(Axi,ur,&rAi); /* r*A*i */
819:   VecDotBegin(Axi,ui,&iAi); /* i*A*i */
820:   VecDotBegin(Bxr,ur,&rBr); /* r*B*r */
821:   VecDotBegin(Bxr,ui,&iBr); /* i*B*r */
822:   VecDotBegin(Bxi,ur,&rBi); /* r*B*i */
823:   VecDotBegin(Bxi,ui,&iBi); /* i*B*i */
824:   VecDotEnd(Axr,ur,&rAr); /* r*A*r */
825:   VecDotEnd(Axr,ui,&iAr); /* i*A*r */
826:   VecDotEnd(Axi,ur,&rAi); /* r*A*i */
827:   VecDotEnd(Axi,ui,&iAi); /* i*A*i */
828:   VecDotEnd(Bxr,ur,&rBr); /* r*B*r */
829:   VecDotEnd(Bxr,ui,&iBr); /* i*B*r */
830:   VecDotEnd(Bxi,ur,&rBi); /* r*B*i */
831:   VecDotEnd(Bxi,ui,&iBi); /* i*B*i */
832:   b0 = rAr+iAi; /* rAr+iAi */
833:   b2 = rAi-iAr; /* rAi-iAr */
834:   b4 = rBr+iBi; /* rBr+iBi */
835:   b6 = rBi-iBr; /* rBi-iBr */
836:   b7 = b4*b4 + b6*b6; /* k */
837:   *eigr = (b0*b4 + b2*b6) / b7; /* eig_r */
838:   *eigi = (b2*b4 - b0*b6) / b7; /* eig_i */
839:   return(0);
840: }

844: PETSC_STATIC_INLINE PetscErrorCode dvd_compute_n_rr(PetscInt i_s,PetscInt n,PetscScalar *eigr,PetscScalar *eigi,Vec *u,Vec *Ax,Vec *Bx)
845: {
846:   PetscErrorCode  ierr;
847:   PetscInt        i;
848:   PetscScalar     b0,b1;

851:   for (i=0; i<n; i++) {
852: #if !defined(PETSC_USE_COMPLEX)
853:     if (eigi[i_s+i] != 0.0) {
854:       PetscScalar eigr0=0.0,eigi0=0.0;
855:       dvd_complex_rayleigh_quotient(u[i],u[i+1],Ax[i],Ax[i+1],Bx[i],Bx[i+1],&eigr0,&eigi0);
856:       if (PetscAbsScalar(eigr[i_s+i]-eigr0)/PetscAbsScalar(eigr[i_s+i]) > 1e-10 || PetscAbsScalar(eigi[i_s+i]-eigi0)/PetscAbsScalar(eigi[i_s+i]) > 1e-10) {
857:         PetscInfo4(u[0], "The eigenvalue %g+%g is far from its Rayleigh quotient value %g+%g\n",(double)eigr[i_s+i],(double)eigi[i_s+i],(double)eigr0,(double)eigi0);
858:       }
859:       i++;
860:     } else
861: #endif
862:     {
863:       VecDotBegin(Ax[i],u[i],&b0);
864:       VecDotBegin(Bx[i],u[i],&b1);
865:       VecDotEnd(Ax[i],u[i],&b0);
866:       VecDotEnd(Bx[i],u[i],&b1);
867:       b0 = b0/b1;
868:       if (PetscAbsScalar(eigr[i_s+i]-b0)/PetscAbsScalar(eigr[i_s+i]) > 1e-10) {
869:         PetscInfo4(u[0], "The eigenvalue %g+%g is far from its Rayleigh quotient value %g+%g\n", (double)PetscRealPart(eigr[i_s+i]),(double)PetscImaginaryPart(eigr[i_s+i]),(double)PetscRealPart(b0),(double)PetscImaginaryPart(b0));
870:       }
871:     }
872:   }
873:   return(0);
874: }

878: /*
879:   Compute: u <- X, v <- K*(theta[0]*A+theta[1]*B)*X,
880:   kr <- K^{-1}*(A-eig*B)*X, being X <- V*pX[i_s..i_e-1], Y <- W*pY[i_s..i_e-1]
881:   where
882:   pX,pY, the right and left eigenvectors of the projected system
883:   ld, the leading dimension of pX and pY
884: */
885: static PetscErrorCode dvd_improvex_jd_proj_uv_KZX(dvdDashboard *d,PetscInt i_s,PetscInt i_e,Vec *u,Vec *v,Vec *kr,PetscScalar *theta,PetscScalar *thetai,PetscScalar *pX,PetscScalar *pY,PetscInt ld)
886: {
887:   PetscErrorCode  ierr;
888:   PetscInt        n = i_e-i_s,i;
889:   PetscScalar     *b;
890:   Vec             *Ax,*Bx,*r;
891:   Mat             M;
892:   BV              X;

895:   BVDuplicateResize(d->eps->V,4,&X);
896:   MatCreateSeqDense(PETSC_COMM_SELF,4,4,NULL,&M);
897:   /* u <- X(i) */
898:   dvd_improvex_compute_X(d,i_s,i_e,u,pX,ld);

900:   /* v <- theta[0]A*u + theta[1]*B*u */

902:   /* Bx <- B*X(i) */
903:   Bx = kr;
904:   if (d->BX) {
905:     for (i=i_s; i<i_e; ++i) {
906:       BVMultVec(d->BX,1.0,0.0,Bx[i-i_s],&pX[ld*i]);
907:     }
908:   } else {
909:     for (i=0;i<n;i++) {
910:       if (d->B) {
911:         MatMult(d->B, u[i], Bx[i]);
912:       } else {
913:         VecCopy(u[i], Bx[i]);
914:       }
915:     }
916:   }

918:   /* Ax <- A*X(i) */
919:   SlepcVecPoolGetVecs(d->auxV,i_e-i_s,&r);
920:   Ax = r;
921:   for (i=i_s; i<i_e; ++i) {
922:     BVMultVec(d->AX,1.0,0.0,Ax[i-i_s],&pX[ld*i]);
923:   }

925:   /* v <- Y(i) */
926:   for (i=i_s; i<i_e; ++i) {
927:     BVMultVec(d->W?d->W:d->eps->V,1.0,0.0,v[i-i_s],&pY[ld*i]);
928:   }

930:   /* Recompute the eigenvalue */
931:   dvd_compute_n_rr(i_s,n,d->eigr,d->eigi,v,Ax,Bx);

933:   for (i=0;i<n;i++) {
934: #if !defined(PETSC_USE_COMPLEX)
935:     if (d->eigi[i_s+i] != 0.0) {
936:       /* [r_i r_i+1 kr_i kr_i+1]*= [ theta_2i'    0            1        0
937:                                        0         theta_2i'     0        1
938:                                      theta_2i+1 -thetai_i   -eigr_i -eigi_i
939:                                      thetai_i    theta_2i+1  eigi_i -eigr_i ] */
940:       MatDenseGetArray(M,&b);
941:       b[0] = b[5] = PetscConj(theta[2*i]);
942:       b[2] = b[7] = -theta[2*i+1];
943:       b[6] = -(b[3] = thetai[i]);
944:       b[1] = b[4] = 0.0;
945:       b[8] = b[13] = 1.0/d->nX[i_s+i];
946:       b[10] = b[15] = -d->eigr[i_s+i]/d->nX[i_s+i];
947:       b[14] = -(b[11] = d->eigi[i_s+i]/d->nX[i_s+i]);
948:       b[9] = b[12] = 0.0;
949:       MatDenseRestoreArray(M,&b);
950:       BVInsertVec(X,0,Ax[i]);
951:       BVInsertVec(X,1,Ax[i+1]);
952:       BVInsertVec(X,2,Bx[i]);
953:       BVInsertVec(X,3,Bx[i+1]);
954:       BVSetActiveColumns(X,0,4);
955:       BVMultInPlace(X,M,0,4);
956:       BVCopyVec(X,0,Ax[i]);
957:       BVCopyVec(X,1,Ax[i+1]);
958:       BVCopyVec(X,2,Bx[i]);
959:       BVCopyVec(X,3,Bx[i+1]);
960:       i++;
961:     } else
962: #endif
963:     {
964:       /* [Ax_i Bx_i]*= [ theta_2i'    1/nX_i
965:                         theta_2i+1  -eig_i/nX_i ] */
966:       MatDenseGetArray(M,&b);
967:       b[0] = PetscConj(theta[i*2]);
968:       b[1] = theta[i*2+1];
969:       b[4] = 1.0/d->nX[i_s+i];
970:       b[5] = -d->eigr[i_s+i]/d->nX[i_s+i];
971:       MatDenseRestoreArray(M,&b);
972:       BVInsertVec(X,0,Ax[i]);
973:       BVInsertVec(X,1,Bx[i]);
974:       BVSetActiveColumns(X,0,2);
975:       BVMultInPlace(X,M,0,2);
976:       BVCopyVec(X,0,Ax[i]);
977:       BVCopyVec(X,1,Bx[i]);
978:     }
979:   }
980:   for (i=0; i<n; i++) d->nX[i_s+i] = 1.0;

982:   /* v <- K^{-1} r = K^{-1}(theta_2i'*Ax + theta_2i+1*Bx) */
983:   for (i=0;i<n;i++) {
984:     d->improvex_precond(d,i_s+i,r[i],v[i]);
985:   }
986:   SlepcVecPoolRestoreVecs(d->auxV,i_e-i_s,&r);

988:   /* kr <- P*(Ax - eig_i*Bx) */
989:   d->calcpairs_proj_res(d,i_s,i_e,kr);
990:   BVDestroy(&X);
991:   MatDestroy(&M);
992:   return(0);
993: }

997: /*
998:   Compute: u <- K^{-1}*X, v <- X,
999:   kr <- K^{-1}*(A-eig*B)*X, being X <- V*pX[i_s..i_e-1]
1000:   where
1001:   pX,pY, the right and left eigenvectors of the projected system
1002:   ld, the leading dimension of pX and pY
1003: */
1004: static PetscErrorCode dvd_improvex_jd_proj_uv_KXX(dvdDashboard *d,PetscInt i_s,PetscInt i_e,Vec *u,Vec *v,Vec *kr,PetscScalar *theta,PetscScalar *thetai,PetscScalar *pX,PetscScalar *pY,PetscInt ld)
1005: {
1006:   PetscErrorCode  ierr;
1007:   PetscInt        n = i_e - i_s,i;
1008:   PetscScalar     *b;
1009:   Vec             *Ax,*Bx,*r;
1010:   Mat             M;
1011:   BV              X;

1014:   BVDuplicateResize(d->eps->V,4,&X);
1015:   MatCreateSeqDense(PETSC_COMM_SELF,4,2,NULL,&M);
1016:   /* [v u] <- X(i) Y(i) */
1017:   dvd_improvex_compute_X(d,i_s,i_e,v,pX,ld);
1018:   for (i=i_s; i<i_e; ++i) {
1019:     BVMultVec(d->W?d->W:d->eps->V,1.0,0.0,u[i-i_s],&pY[ld*i]);
1020:   }

1022:   /* Bx <- B*X(i) */
1023:   SlepcVecPoolGetVecs(d->auxV,i_e-i_s,&r);
1024:   Bx = r;
1025:   if (d->BX) {
1026:     for (i=i_s; i<i_e; ++i) {
1027:       BVMultVec(d->BX,1.0,0.0,Bx[i-i_s],&pX[ld*i]);
1028:     }
1029:   } else {
1030:     if (d->B) {
1031:       for (i=0;i<n;i++) {
1032:         MatMult(d->B,v[i],Bx[i]);
1033:       }
1034:     } else Bx = v;
1035:   }

1037:   /* Ax <- A*X(i) */
1038:   Ax = kr;
1039:   for (i=i_s; i<i_e; ++i) {
1040:     BVMultVec(d->AX,1.0,0.0,Ax[i-i_s],&pX[ld*i]);
1041:   }

1043:   /* Recompute the eigenvalue */
1044:   dvd_compute_n_rr(i_s,n,d->eigr,d->eigi,u,Ax,Bx);

1046:   for (i=0;i<n;i++) {
1047:     if (d->eigi[i_s+i] == 0.0) {
1048:       /* kr <- Ax -eig*Bx */
1049:       VecAXPBY(kr[i],-d->eigr[i_s+i]/d->nX[i_s+i],1.0/d->nX[i_s+i],Bx[i]);
1050:     } else {
1051:       /* [kr_i kr_i+1 r_i r_i+1]*= [   1        0
1052:                                        0        1
1053:                                     -eigr_i -eigi_i
1054:                                      eigi_i -eigr_i] */
1055:       MatDenseGetArray(M,&b);
1056:       b[0] = b[5] = 1.0/d->nX[i_s+i];
1057:       b[2] = b[7] = -d->eigr[i_s+i]/d->nX[i_s+i];
1058:       b[6] = -(b[3] = d->eigi[i_s+i]/d->nX[i_s+i]);
1059:       b[1] = b[4] = 0.0;
1060:       MatDenseRestoreArray(M,&b);
1061:       BVInsertVec(X,0,kr[i]);
1062:       BVInsertVec(X,1,kr[i+1]);
1063:       BVInsertVec(X,2,r[i]);
1064:       BVInsertVec(X,3,r[i+1]);
1065:       BVSetActiveColumns(X,0,4);
1066:       BVMultInPlace(X,M,0,2);
1067:       BVCopyVec(X,0,kr[i]);
1068:       BVCopyVec(X,1,kr[i+1]);
1069:       i++;
1070:     }
1071:   }
1072:   for (i=0; i<n; i++) d->nX[i_s+i] = 1.0;

1074:   /* kr <- P*kr */
1075:   d->calcpairs_proj_res(d,i_s,i_e,r);
1076:   SlepcVecPoolRestoreVecs(d->auxV,i_e-i_s,&r);

1078:   /* u <- K^{-1} X(i) */
1079:   for (i=0;i<n;i++) {
1080:     d->improvex_precond(d,i_s+i,v[i],u[i]);
1081:   }
1082:   return(0);
1083: }
