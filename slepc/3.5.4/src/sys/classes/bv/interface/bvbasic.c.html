 31: /*@C
 32:    BVSetType - Selects the type for the BV object.

 34:    Logically Collective on BV

 36:    Input Parameter:
 37: +  bv   - the basis vectors context
 38: -  type - a known type

 40:    Options Database Key:
 41: .  -bv_type <type> - Sets BV type

 43:    Level: intermediate

 45: .seealso: BVGetType()
 46: @*/
 47: PetscErrorCode BVSetType(BV bv,BVType type)
 48: {
 49:   PetscErrorCode ierr,(*r)(BV);
 50:   PetscBool      match;


 56:   PetscObjectTypeCompare((PetscObject)bv,type,&match);
 57:   if (match) return(0);

 59:    PetscFunctionListFind(BVList,type,&r);
 60:   if (!r) SETERRQ1(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_UNKNOWN_TYPE,"Unable to find requested BV type %s",type);

 62:   if (bv->ops->destroy) { (*bv->ops->destroy)(bv); }
 63:   PetscMemzero(bv->ops,sizeof(struct _BVOps));

 65:   PetscObjectChangeTypeName((PetscObject)bv,type);
 66:   if (bv->n < 0 && bv->N < 0) {
 67:     bv->ops->create = r;
 68:   } else {
 69:     PetscLogEventBegin(BV_Create,bv,0,0,0);
 70:     (*r)(bv);
 71:     PetscLogEventEnd(BV_Create,bv,0,0,0);
 72:   }
 73:   return(0);
 74: }

104: /*@
105:    BVSetSizes - Sets the local and global sizes, and the number of columns.

107:    Collective on BV

109:    Input Parameters:
110: +  bv - the basis vectors
111: .  n  - the local size (or PETSC_DECIDE to have it set)
112: .  N  - the global size (or PETSC_DECIDE)
113: -  m  - the number of columns

115:    Notes:
116:    n and N cannot be both PETSC_DECIDE.
117:    If one processor calls this with N of PETSC_DECIDE then all processors must,
118:    otherwise the program will hang.

120:    Level: beginner

122: .seealso: BVSetSizesFromVec(), BVGetSizes(), BVResize()
123: @*/
124: PetscErrorCode BVSetSizes(BV bv,PetscInt n,PetscInt N,PetscInt m)
125: {
127:   PetscInt       ma;

133:   if (N >= 0 && n > N) SETERRQ2(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"Local size %D cannot be larger than global size %D",n,N);
134:   if (m <= 0) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"Number of columns %D must be positive",m);
135:   if ((bv->n >= 0 || bv->N >= 0) && (bv->n != n || bv->N != N)) SETERRQ4(PETSC_COMM_SELF,PETSC_ERR_SUP,"Cannot change/reset vector sizes to %D local %D global after previously setting them to %D local %D global",n,N,bv->n,bv->N);
136:   if (bv->m > 0 && bv->m != m) SETERRQ2(PETSC_COMM_SELF,PETSC_ERR_SUP,"Cannot change the number of columns to %D after previously setting it to %D; use BVResize()",m,bv->m);
137:   bv->n = n;
138:   bv->N = N;
139:   bv->m = m;
140:   bv->k = m;
141:   if (!bv->t) {  /* create template vector and get actual dimensions */
142:     VecCreate(PetscObjectComm((PetscObject)bv),&bv->t);
143:     VecSetSizes(bv->t,bv->n,bv->N);
144:     VecSetFromOptions(bv->t);
145:     VecGetSize(bv->t,&bv->N);
146:     VecGetLocalSize(bv->t,&bv->n);
147:     if (bv->matrix) {  /* check compatible dimensions of user-provided matrix */
148:       MatGetLocalSize(bv->matrix,&ma,NULL);
149:       if (bv->n!=ma) SETERRQ2(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"Local dimension %D does not match that of matrix given at BVSetMatrix %D",bv->n,ma);
150:     }
151:   }
152:   if (bv->ops->create) {
153:     PetscLogEventBegin(BV_Create,bv,0,0,0);
154:     (*bv->ops->create)(bv);
155:     PetscLogEventEnd(BV_Create,bv,0,0,0);
156:     bv->ops->create = 0;
157:   }
158:   return(0);
159: }

165:    Local and global sizes are specified indirectly by passing a template vector.

167:    Collective on BV

169:    Input Parameters:
170: +  bv - the basis vectors
171: .  t  - the template vectors
172: -  m  - the number of columns

174:    Level: beginner

176: .seealso: BVSetSizes(), BVGetSizes(), BVResize()
177: @*/
178: PetscErrorCode BVSetSizesFromVec(BV bv,Vec t,PetscInt m)
179: {
181:   PetscInt       ma;

188:   if (m <= 0) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"Number of columns %D must be positive",m);
189:   if (bv->t) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_SUP,"Template vector was already set by a previous call to BVSetSizes/FromVec");
190:   VecGetSize(t,&bv->N);
191:   VecGetLocalSize(t,&bv->n);
192:   if (bv->matrix) {  /* check compatible dimensions of user-provided matrix */
193:     MatGetLocalSize(bv->matrix,&ma,NULL);
194:     if (bv->n!=ma) SETERRQ2(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"Local dimension %D does not match that of matrix given at BVSetMatrix %D",bv->n,ma);

196:   bv->m = m;
197:   bv->k = m;
198:   bv->t = t;
199:   PetscObjectReference((PetscObject)t);
200:   if (bv->ops->create) {
201:     (*bv->ops->create)(bv);
202:     bv->ops->create = 0;
203:   }
204:   return(0);
205: }

209: /*@
210:    BVGetSizes - Returns the local and global sizes, and the number of columns.

212:    Not Collective

214:    Input Parameter:
215: .  bv - the basis vectors

217:    Output Parameters:
218: +  n  - the local size
219: .  N  - the global size
220: -  m  - the number of columns

222:    Note:
223:    Normal usage requires that bv has already been given its sizes, otherwise
224:    the call fails. However, this function can also be used to determine if
225:    a BV object has been initialized completely (sizes and type). For this,
226:    call with n=NULL and N=NULL, then a return value of m=0 indicates that
227:    the BV object is not ready for use yet.

229:    Level: beginner

231: .seealso: BVSetSizes(), BVSetSizesFromVec()
232: @*/
233: PetscErrorCode BVGetSizes(BV bv,PetscInt *n,PetscInt *N,PetscInt *m)
234: {
236:   if (!bv) {
237:     if (m && !n && !N) *m = 0;
238:     return(0);
239:   }
241:   if (n || N) BVCheckSizes(bv,1);
242:   if (n) *n = bv->n;
243:   if (N) *N = bv->N;
244:   if (m) *m = bv->m;
245:   if (m && !n && !N && !((PetscObject)bv)->type_name) *m = 0;
246:   return(0);
247: }

251: /*@
252:    BVSetNumConstraints - Set the number of constraints.

254:    Logically Collective on BV

256:    Input Parameters:
257: +  V  - basis vectors
258: -  nc - number of constraints

260:    Notes:
261:    This function sets the number of constraints to nc and marks all remaining
262:    columns as regular. Normal user would call BVInsertConstraints() instead.

264:    If nc is smaller than the previously set value, then some of the constraints
265:    are discarded. In particular, using nc=0 removes all constraints preserving
266:    the content of regular columns.

268:    Level: developer

270: .seealso: BVInsertConstraints()
271: @*/
272: PetscErrorCode BVSetNumConstraints(BV V,PetscInt nc)
273: {
275:   PetscInt       total,diff,i;
276:   Vec            x,y;

281:   if (nc<0) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_ARG_OUTOFRANGE,"Number of constraints (given %D) cannot be negative",nc);
283:   BVCheckSizes(V,1);
284:   if (V->ci[0]!=-V->nc-1 || V->ci[1]!=-V->nc-1) SETERRQ(PetscObjectComm((PetscObject)V),PETSC_ERR_SUP,"Cannot call BVSetNumConstraints after BVGetColumn");

286:   diff = nc-V->nc;
287:   if (!diff) return(0);
288:   total = V->nc+V->m;
289:   if (total-nc<=0) SETERRQ(PetscObjectComm((PetscObject)V),PETSC_ERR_ARG_OUTOFRANGE,"Not enough columns for the given nc value");
290:   if (diff<0) {  /* lessen constraints, shift contents of BV */
291:     for (i=0;i<V->m;i++) {
292:       BVGetColumn(V,i,&x);
293:       BVGetColumn(V,i+diff,&y);
294:       VecCopy(x,y);
295:       BVRestoreColumn(V,i,&x);
296:       BVRestoreColumn(V,i+diff,&y);
297:     }
298:   }
299:   V->nc = nc;
300:   V->ci[0] = -V->nc-1;
301:   V->ci[1] = -V->nc-1;
302:   V->l = 0;
303:   V->m = total-nc;
304:   V->k = V->m;
305:   PetscObjectStateIncrease((PetscObject)V);
306:   return(0);
307: }

337: /*@
338:    BVResize - Change the number of columns.

340:    Collective on BV

342:    Input Parameters:
343: +  bv   - the basis vectors
344: .  m    - the new number of columns
345: -  copy - a flag indicating whether current values should be kept

347:    Note:
348:    Internal storage is reallocated. If the copy flag is set to true, then
349:    the contents are copied to the leading part of the new space.

351:    Level: advanced

353: .seealso: BVSetSizes(), BVSetSizesFromVec()
354: @*/
355: PetscErrorCode BVResize(BV bv,PetscInt m,PetscBool copy)
356: {
358:   PetscReal      *omega;
359:   PetscInt       i;

366:   if (m <= 0) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"Number of columns %D must be positive",m);
367:   if (bv->nc) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_ARG_WRONGSTATE,"Cannot resize a BV with constraints");
368:   if (bv->m == m) return(0);

370:   PetscLogEventBegin(BV_Create,bv,0,0,0);
371:   (*bv->ops->resize)(bv,m,copy);
372:   PetscFree2(bv->h,bv->c);
373:   if (bv->omega) {
374:     PetscMalloc1(m,&omega);
375:     PetscLogObjectMemory((PetscObject)bv,m*sizeof(PetscReal));
376:     for (i=0;i<m;i++) omega[i] = 1.0;
377:     if (copy) {
378:       PetscMemcpy(omega,bv->omega,PetscMin(m,bv->m)*sizeof(PetscReal));
379:     }
380:     PetscFree(bv->omega);
381:     bv->omega = omega;
382:   }
383:   bv->m = m;
384:   bv->k = PetscMin(bv->k,m);
385:   bv->l = PetscMin(bv->l,m);
386:   PetscLogEventEnd(BV_Create,bv,0,0,0);
387:   return(0);
388: }

392: /*@
393:    BVSetActiveColumns - Specify the columns that will be involved in operations.

395:    Logically Collective on BV

397:    Input Parameters:
398: +  bv - the basis vectors context
399: .  l  - number of leading columns
400: -  k  - number of active columns

402:    Notes:
403:    In operations such as BVMult() or BVDot(), only the first k columns are
404:    considered. This is useful when the BV is filled from left to right, so
405:    the last m-k columns do not have relevant information.

407:    Also in operations such as BVMult() or BVDot(), the first l columns are
408:    normally not included in the computation. See the manpage of each
409:    operation.

411:    In orthogonalization operations, the first l columns are treated
412:    differently: they participate in the orthogonalization but the computed
413:    coefficients are not stored.

415:    Level: intermediate

417: .seealso: BVGetActiveColumns(), BVSetSizes()
418: @*/
419: PetscErrorCode BVSetActiveColumns(BV bv,PetscInt l,PetscInt k)
420: {
425:   BVCheckSizes(bv,1);
426:   if (k==PETSC_DECIDE || k==PETSC_DEFAULT) {
427:     bv->k = bv->m;
428:   } else {
429:     if (k<0 || k>bv->m) SETERRQ(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_OUTOFRANGE,"Illegal value of k. Must be between 0 and m");
430:     bv->k = k;
431:   }
432:   if (l==PETSC_DECIDE || l==PETSC_DEFAULT) {
433:     bv->l = 0;
434:   } else {
435:     if (l<0 || l>bv->k) SETERRQ(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_OUTOFRANGE,"Illegal value of l. Must be between 0 and k");
436:     bv->l = l;
437:   }
438:   return(0);
439: }

470: /*@
471:    BVSetMatrix - Specifies the inner product to be used in orthogonalization.

473:    Collective on BV

475:    Input Parameters:
476: +  bv    - the basis vectors context
477: .  B     - a symmetric matrix (may be NULL)
478: -  indef - a flag indicating if the matrix is indefinite

480:    Notes:
481:    This is used to specify a non-standard inner product, whose matrix
482:    representation is given by B. Then, all inner products required during
483:    orthogonalization are computed as (x,y)_B=y^H*B*x rather than the
484:    standard form (x,y)=y^H*x.

486:    Matrix B must be real symmetric (or complex Hermitian). A genuine inner
487:    product requires that B is also positive (semi-)definite. However, we
488:    also allow for an indefinite B (setting indef=PETSC_TRUE), in which
489:    case the orthogonalization uses an indefinite inner product.

491:    This affects operations BVDot(), BVNorm(), BVOrthogonalize(), and variants.

493:    Setting B=NULL has the same effect as if the identity matrix was passed.

495:    Level: advanced

497: .seealso: BVGetMatrix(), BVDot(), BVNorm(), BVOrthogonalize()
498: @*/
499: PetscErrorCode BVSetMatrix(BV bv,Mat B,PetscBool indef)
500: {
502:   PetscInt       m,n;

507:   if (B) {
509:     MatGetLocalSize(B,&m,&n);
510:     if (m!=n) SETERRQ(PETSC_COMM_SELF,PETSC_ERR_ARG_SIZ,"Matrix must be square");
511:     if (bv->m && bv->n!=n) SETERRQ2(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"Mismatching local dimension BV %D, Mat %D",bv->n,n);
512:   }
513:   MatDestroy(&bv->matrix);
514:   if (B) PetscObjectReference((PetscObject)B);
515:   bv->matrix = B;
516:   bv->indef  = indef;
517:   if (B && !bv->Bx) {
518:     MatGetVecs(B,&bv->Bx,NULL);
519:     PetscLogObjectParent((PetscObject)bv,(PetscObject)bv->Bx);
520:   }
521:   return(0);
522: }

592: /*@
593:    BVSetSignature - Sets the signature matrix to be used in orthogonalization.

595:    Logically Collective on BV

597:    Input Parameter:
598: +  bv    - the basis vectors context
599: -  omega - a vector representing the diagonal of the signature matrix

601:    Note:
602:    The signature matrix Omega = V'*B*V is relevant only for an indefinite B.

604:    Level: developer

606: .seealso: BVSetMatrix(), BVGetSignature()
607: @*/
608: PetscErrorCode BVSetSignature(BV bv,Vec omega)
609: {
611:   PetscInt       i,n;
612:   PetscScalar    *pomega;

616:   BVCheckSizes(bv,1);

620:   VecGetSize(omega,&n);
621:   if (n!=bv->k) SETERRQ2(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_SIZ,"Vec argument has %D elements, should be %D",n,bv->k);
622:   BV_AllocateSignature(bv);
623:   if (bv->indef) {
624:     VecGetArray(omega,&pomega);
625:     for (i=0;i<n;i++) bv->omega[bv->nc+i] = PetscRealPart(pomega[i]);
626:     VecRestoreArray(omega,&pomega);
627:   } else {
628:     PetscInfo(bv,"Ignoring signature because BV is not indefinite\n");
629:   }
630:   return(0);
631: }

635: /*@
636:    BVGetSignature - Retrieves the signature matrix from last orthogonalization.

638:    Not collective

640:    Input Parameter:
641: .  bv    - the basis vectors context

643:    Output Parameter:
644: .  omega - a vector representing the diagonal of the signature matrix

646:    Note:
647:    The signature matrix Omega = V'*B*V is relevant only for an indefinite B.

649:    Level: developer

651: .seealso: BVSetMatrix(), BVSetSignature()
652: @*/
653: PetscErrorCode BVGetSignature(BV bv,Vec omega)
654: {
656:   PetscInt       i,n;
657:   PetscScalar    *pomega;

661:   BVCheckSizes(bv,1);

665:   VecGetSize(omega,&n);
666:   if (n!=bv->k) SETERRQ2(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_SIZ,"Vec argument has %D elements, should be %D",n,bv->k);
667:   if (bv->indef && bv->omega) {
668:     VecGetArray(omega,&pomega);
669:     for (i=0;i<n;i++) pomega[i] = bv->omega[bv->nc+i];
670:     VecRestoreArray(omega,&pomega);
671:   } else {
672:     VecSet(omega,1.0);
673:   }
674:   return(0);
675: }

679: /*@
680:    BVSetFromOptions - Sets BV options from the options database.

682:    Collective on BV

684:    Input Parameter:
685: .  bv - the basis vectors context

687:    Level: beginner
688: @*/
689: PetscErrorCode BVSetFromOptions(BV bv)
690: {
692:   char           type[256];
693:   PetscBool      flg;
694:   PetscReal      r;
695:   PetscInt       i,j;
696:   const char     *orth_list[2] = {"cgs","mgs"};
697:   const char     *ref_list[3] = {"ifneeded","never","always"};

701:   if (!BVRegisterAllCalled) { BVRegisterAll(); }
702:   PetscObjectOptionsBegin((PetscObject)bv);
703:     PetscOptionsFList("-bv_type","Basis Vectors type","BVSetType",BVList,(char*)(((PetscObject)bv)->type_name?((PetscObject)bv)->type_name:BVSVEC),type,256,&flg);
704:     if (flg) {
705:       BVSetType(bv,type);
706:     }
707:     /*
708:       Set the type if it was never set.
709:     */
710:     if (!((PetscObject)bv)->type_name) {
711:       BVSetType(bv,BVSVEC);
712:     }

714:     i = bv->orthog_type;
715:     PetscOptionsEList("-bv_orthog_type","Orthogonalization method","BVSetOrthogonalization",orth_list,2,orth_list[i],&i,NULL);
716:     j = bv->orthog_ref;
717:     PetscOptionsEList("-bv_orthog_refine","Iterative refinement mode during orthogonalization","BVSetOrthogonalization",ref_list,3,ref_list[j],&j,NULL);
718:     r = bv->orthog_eta;
719:     PetscOptionsReal("-bv_orthog_eta","Parameter of iterative refinement during orthogonalization","BVSetOrthogonalization",r,&r,NULL);
720:     BVSetOrthogonalization(bv,(BVOrthogType)i,(BVOrthogRefineType)j,r);

722:     if (bv->ops->setfromoptions) {
723:       (*bv->ops->setfromoptions)(bv);
724:     }
725:     PetscObjectProcessOptionsHandlers((PetscObject)bv);
726:   PetscOptionsEnd();
727:   return(0);
728: }

732: /*@
733:    BVSetOrthogonalization - Specifies the type of orthogonalization technique
734:    to be used (classical or modified Gram-Schmidt with or without refinement).

736:    Logically Collective on BV

738:    Input Parameters:
739: +  bv     - the basis vectors context
740: .  type   - the type of orthogonalization technique
741: .  refine - type of refinement
742: -  eta    - parameter for selective refinement

744:    Options Database Keys:
745: +  -bv_orthog_type <type> - Where <type> is cgs for Classical Gram-Schmidt orthogonalization
746:                          (default) or mgs for Modified Gram-Schmidt orthogonalization
747: .  -bv_orthog_refine <ref> - Where <ref> is one of never, ifneeded (default) or always
748: -  -bv_orthog_eta <eta> -  For setting the value of eta

750:    Notes:
751:    The default settings work well for most problems.

753:    The parameter eta should be a real value between 0 and 1 (or PETSC_DEFAULT).
754:    The value of eta is used only when the refinement type is "ifneeded".

756:    When using several processors, MGS is likely to result in bad scalability.

758:    Level: advanced

760: .seealso: BVOrthogonalizeColumn(), BVGetOrthogonalization(), BVOrthogType, BVOrthogRefineType
761: @*/
762: PetscErrorCode BVSetOrthogonalization(BV bv,BVOrthogType type,BVOrthogRefineType refine,PetscReal eta)
763: {
769:   switch (type) {
770:     case BV_ORTHOG_CGS:
771:     case BV_ORTHOG_MGS:
772:       bv->orthog_type = type;
773:       break;
774:     default:
775:       SETERRQ(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_WRONG,"Unknown orthogonalization type");
776:   }
777:   switch (refine) {
778:     case BV_ORTHOG_REFINE_NEVER:
779:     case BV_ORTHOG_REFINE_IFNEEDED:
780:     case BV_ORTHOG_REFINE_ALWAYS:
781:       bv->orthog_ref = refine;
782:       break;
783:     default:
784:       SETERRQ(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_WRONG,"Unknown refinement type");
785:   }
786:   if (eta == PETSC_DEFAULT) {
787:     bv->orthog_eta = 0.7071;
788:   } else {
789:     if (eta <= 0.0 || eta > 1.0) SETERRQ(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_OUTOFRANGE,"Invalid eta value");
790:     bv->orthog_eta = eta;
791:   }
792:   return(0);
793: }

805:    Output Parameter:

807: .  refine - type of refinement

826: /*@
827:    BVGetColumn - Returns a Vec object that contains the entries of the
828:    requested column of the basis vectors object.

830:    Logically Collective on BV

832:    Input Parameters:
833: +  bv - the basis vectors context
834: -  j  - the index of the requested column

836:    Output Parameter:
837: .  v  - vector containing the jth column

839:    Notes:
840:    The returned Vec must be seen as a reference (not a copy) of the BV
841:    column, that is, modifying the Vec will change the BV entries as well.

843:    The returned Vec must not be destroyed. BVRestoreColumn() must be
844:    called when it is no longer needed. At most, two columns can be fetched,
845:    that is, this function can only be called twice before the corresponding
846:    BVRestoreColumn() is invoked.

848:    A negative index j selects the i-th constraint, where i=-j. Constraints
849:    should not be modified.

851:    Level: beginner

853: .seealso: BVRestoreColumn(), BVInsertConstraints()
854: @*/
855: PetscErrorCode BVGetColumn(BV bv,PetscInt j,Vec *v)
856: {
858:   PetscInt       l;

863:   BVCheckSizes(bv,1);
865:   if (j<0 && -j>bv->nc) SETERRQ2(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_OUTOFRANGE,"You requested constraint %D but only %D are available",-j,bv->nc);
866:   if (j>=bv->m) SETERRQ2(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_OUTOFRANGE,"You requested column %D but only %D are available",j,bv->m);
867:   if (j==bv->ci[0] || j==bv->ci[1]) SETERRQ1(PetscObjectComm((PetscObject)bv),PETSC_ERR_SUP,"Column %D already fetched in a previous call to BVGetColumn",j);
868:   l = BVAvailableVec;
869:   if (l==-1) SETERRQ(PetscObjectComm((PetscObject)bv),PETSC_ERR_SUP,"Too many requested columns; you must call BVReleaseColumn for one of the previously fetched columns");
870:   (*bv->ops->getcolumn)(bv,j,v);
871:   bv->ci[l] = j;
872:   PetscObjectStateGet((PetscObject)bv->cv[l],&bv->st[l]);
873:   PetscObjectGetId((PetscObject)bv->cv[l],&bv->id[l]);
874:   *v = bv->cv[l];
875:   return(0);
876: }

880: /*@
881:    BVRestoreColumn - Restore a column obtained with BVGetColumn().

883:    Logically Collective on BV

885:    Input Parameters:
886: +  bv - the basis vectors context
887: .  j  - the index of the column
888: -  v  - vector obtained with BVGetColumn()

890:    Note:
891:    The arguments must match the corresponding call to BVGetColumn().

893:    Level: beginner

895: .seealso: BVGetColumn()
896: @*/
897: PetscErrorCode BVRestoreColumn(BV bv,PetscInt j,Vec *v)
898: {
899:   PetscErrorCode   ierr;
900:   PetscObjectId    id;
901:   PetscObjectState st;
902:   PetscInt         l;

907:   BVCheckSizes(bv,1);
911:   if (j<0 && -j>bv->nc) SETERRQ2(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_OUTOFRANGE,"You requested constraint %D but only %D are available",-j,bv->nc);
912:   if (j>=bv->m) SETERRQ2(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_OUTOFRANGE,"You requested column %D but only %D are available",j,bv->m);
913:   if (j!=bv->ci[0] && j!=bv->ci[1]) SETERRQ1(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_WRONG,"Column %D has not been fetched with a call to BVGetColumn",j);
914:   l = (j==bv->ci[0])? 0: 1;
915:   PetscObjectGetId((PetscObject)*v,&id);
916:   if (id!=bv->id[l]) SETERRQ(PetscObjectComm((PetscObject)bv),PETSC_ERR_ARG_WRONG,"Argument 3 is not the same Vec that was obtained with BVGetColumn");
917:   PetscObjectStateGet((PetscObject)*v,&st);
918:   if (st!=bv->st[l]) {
919:     PetscObjectStateIncrease((PetscObject)bv);
920:   }
921:   if (bv->ops->restorecolumn) {
922:     (*bv->ops->restorecolumn)(bv,j,v);
923:   } else bv->cv[l] = NULL;
924:   bv->ci[l] = -bv->nc-1;
925:   bv->st[l] = -1;
926:   bv->id[l] = 0;
927:   *v = NULL;
928:   return(0);
929: }

978: -  a  - location of pointer to array obtained from BVGetArray()

1037: /*@
1038:    BVDuplicate - Creates a new basis vector object of the same type and
1039:    dimensions as an existing one.

1041:    Collective on BV

1043:    Input Parameter:
1044: .  V - basis vectors context

1046:    Output Parameter:
1047: .  W - location to put the new BV

1049:    Notes:
1050:    The new BV has the same type and dimensions as V, and it shares the same
1051:    template vector. Also, the inner product matrix and orthogonalization
1052:    options are copied.

1054:    BVDuplicate() DOES NOT COPY the entries, but rather allocates storage
1055:    for the new basis vectors. Use BVCopy() to copy the contents.

1057:    Level: intermediate

1059: .seealso: BVCreate(), BVSetSizesFromVec(), BVCopy()
1060: @*/
1061: PetscErrorCode BVDuplicate(BV V,BV *W)
1062: {

1068:   BVCheckSizes(V,1);
1070:   BVCreate(PetscObjectComm((PetscObject)V),W);
1071:   BVSetSizesFromVec(*W,V->t,V->m);
1072:   BVSetType(*W,((PetscObject)V)->type_name);
1073:   BVSetMatrix(*W,V->matrix,V->indef);
1074:   BVSetOrthogonalization(*W,V->orthog_type,V->orthog_ref,V->orthog_eta);
1075:   PetscObjectStateIncrease((PetscObject)*W);
1076:   return(0);
1077: }

1123: /*@
1124:    BVCopy - Copies a basis vector object into another one, W <- V.

1126:    Logically Collective on BV

1128:    Input Parameter:
1129: .  V - basis vectors context

1131:    Output Parameter:
1132: .  W - the copy

1134:    Note:
1135:    Both V and W must be distributed in the same manner; local copies are
1136:    done. Only active columns (excluding the leading ones) are copied.
1137:    In the destination W, columns are overwritten starting from the leading ones.
1138:    Constraints are not copied.

1140:    Level: beginner

1142: .seealso: BVCopyVec(), BVCopyColumn(), BVDuplicate(), BVSetActiveColumns()
1143: @*/
1144: PetscErrorCode BVCopy(BV V,BV W)
1145: {

1151:   BVCheckSizes(V,1);
1154:   BVCheckSizes(W,2);
1156:   if (V->n!=W->n) SETERRQ2(PETSC_COMM_SELF,PETSC_ERR_ARG_INCOMP,"Mismatching local dimension V %D, W %D",V->n,W->n);
1157:   if (V->k-V->l>W->m-W->l) SETERRQ2(PETSC_COMM_SELF,PETSC_ERR_ARG_SIZ,"W has %D non-leading columns, not enough to store %D columns",W->m-W->l,V->k-V->l);
1158:   if (!V->n) return(0);

1160:   PetscLogEventBegin(BV_Copy,V,W,0,0);
1161:   if (V->indef && V->matrix && V->indef==W->indef && V->matrix==W->matrix) {
1162:     /* copy signature */
1163:     BV_AllocateSignature(W);
1164:     PetscMemcpy(W->omega+W->nc+W->l,V->omega+V->nc+V->l,(V->k-V->l)*sizeof(PetscReal));
1165:   }
1166:   (*V->ops->copy)(V,W);
1167:   PetscLogEventEnd(BV_Copy,V,W,0,0);
1168:   return(0);
1169: }

1173: /*@
1174:    BVCopyVec - Copies one of the columns of a basis vectors object into a Vec.

1176:    Logically Collective on BV

1178:    Input Parameter:
1179: +  V - basis vectors context
1180: -  j - the column number to be copied

1182:    Output Parameter:
1183: .  w - the copied column

1185:    Note:
1186:    Both V and w must be distributed in the same manner; local copies are done.

1188:    Level: beginner

1190: .seealso: BVCopy(), BVCopyColumn(), BVInsertVec()
1191: @*/
1192: PetscErrorCode BVCopyVec(BV V,PetscInt j,Vec w)
1193: {
1195:   PetscInt       n,N;
1196:   Vec            z;

1201:   BVCheckSizes(V,1);

1206:   VecGetSize(w,&N);
1207:   VecGetLocalSize(w,&n);
1208:   if (N!=V->N || n!=V->n) SETERRQ4(PetscObjectComm((PetscObject)V),PETSC_ERR_ARG_INCOMP,"Vec sizes (global %D, local %D) do not match BV sizes (global %D, local %D)",N,n,V->N,V->n);

1210:   PetscLogEventBegin(BV_Copy,V,w,0,0);
1211:   BVGetColumn(V,j,&z);
1212:   VecCopy(z,w);
1213:   BVRestoreColumn(V,j,&z);
1214:   PetscLogEventEnd(BV_Copy,V,w,0,0);
1215:   return(0);
1216: }

1220: /*@
1221:    BVCopyColumn - Copies the values from one of the columns to another one.

1223:    Logically Collective on BV

1225:    Input Parameter:
1226: +  V - basis vectors context
1227: .  j - the number of the source column
1228: -  i - the number of the destination column

1230:    Level: beginner

1232: .seealso: BVCopy(), BVCopyVec()
1233: @*/
1234: PetscErrorCode BVCopyColumn(BV V,PetscInt j,PetscInt i)
1235: {
1237:   Vec            z,w;

1242:   BVCheckSizes(V,1);
1245:   if (j==i) return(0);

1247:   PetscLogEventBegin(BV_Copy,V,0,0,0);
1248:   if (V->omega) V->omega[i] = V->omega[j];
1249:   BVGetColumn(V,j,&z);
1250:   BVGetColumn(V,i,&w);
1251:   VecCopy(z,w);
1252:   BVRestoreColumn(V,j,&z);
1253:   BVRestoreColumn(V,i,&w);
1254:   PetscLogEventEnd(BV_Copy,V,0,0,0);
1255:   return(0);
1256: }
