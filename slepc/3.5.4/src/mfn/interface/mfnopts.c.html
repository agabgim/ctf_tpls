 29: /*@
 30:    MFNSetFromOptions - Sets MFN options from the options database.
 31:    This routine must be called before MFNSetUp() if the user is to be
 32:    allowed to set the solver type.

 34:    Collective on MFN

 36:    Input Parameters:
 37: .  mfn - the matrix function context

 39:    Notes:
 40:    To see all options, run your program with the -help option.

 42:    Level: beginner
 43: @*/
 44: PetscErrorCode MFNSetFromOptions(MFN mfn)
 45: {
 46:   PetscErrorCode   ierr;
 47:   char             type[256],monfilename[PETSC_MAX_PATH_LEN];
 48:   PetscBool        flg,flg1,flg2;
 49:   PetscReal        r;
 50:   PetscInt         i;
 51:   PetscViewer      monviewer;

 55:   if (!MFNRegisterAllCalled) { MFNRegisterAll(); }
 56:   PetscObjectOptionsBegin((PetscObject)mfn);
 57:     PetscOptionsFList("-mfn_type","Matrix Function method","MFNSetType",MFNList,(char*)(((PetscObject)mfn)->type_name?((PetscObject)mfn)->type_name:MFNKRYLOV),type,256,&flg);
 58:     if (flg) {
 59:       MFNSetType(mfn,type);
 60:     }
 61:     /*
 62:       Set the type if it was never set.
 63:     */
 64:     if (!((PetscObject)mfn)->type_name) {
 65:       MFNSetType(mfn,MFNKRYLOV);
 66:     }

 68:     PetscOptionsBoolGroupBegin("-mfn_exp","matrix exponential","MFNSetFunction",&flg);
 69:     if (flg) {
 70:       MFNSetFunction(mfn,SLEPC_FUNCTION_EXP);
 71:     }

 73:     PetscOptionsScalar("-mfn_scale","Scale factor","MFNSetScaleFactor",mfn->sfactor,&mfn->sfactor,NULL);

 75:     i = mfn->max_it;
 76:     PetscOptionsInt("-mfn_max_it","Maximum number of iterations","MFNSetTolerances",mfn->max_it,&i,&flg1);
 77:     if (!flg1) i = PETSC_DEFAULT;
 78:     r = mfn->tol;
 79:     PetscOptionsReal("-mfn_tol","Tolerance","MFNSetTolerances",mfn->tol==PETSC_DEFAULT?SLEPC_DEFAULT_TOL:mfn->tol,&r,&flg2);
 80:     if (flg1 || flg2) {
 81:       MFNSetTolerances(mfn,r,i);
 82:     }

 84:     PetscOptionsInt("-mfn_ncv","Number of basis vectors","MFNSetDimensions",mfn->ncv,&i,&flg);
 85:     if (flg) {
 86:       MFNSetDimensions(mfn,i);
 87:     }

 89:     PetscOptionsBool("-mfn_error_if_not_converged","Generate error if solver does not converge","MFNSetErrorIfNotConverged",mfn->errorifnotconverged,&mfn->errorifnotconverged,NULL);

 91:     /* -----------------------------------------------------------------------*/
 92:     /*
 93:       Cancels all monitors hardwired into code before call to MFNSetFromOptions()
 94:     */
 95:     flg = PETSC_FALSE;
 96:     PetscOptionsBool("-mfn_monitor_cancel","Remove any hardwired monitor routines","MFNMonitorCancel",flg,&flg,NULL);
 97:     if (flg) {
 98:       MFNMonitorCancel(mfn);
 99:     }
100:     /*
101:       Prints error estimate at each iteration
102:     */
103:     PetscOptionsString("-mfn_monitor","Monitor error estimate","MFNMonitorSet","stdout",monfilename,PETSC_MAX_PATH_LEN,&flg);
104:     if (flg) {
105:       PetscViewerASCIIOpen(PetscObjectComm((PetscObject)mfn),monfilename,&monviewer);
106:       MFNMonitorSet(mfn,MFNMonitorDefault,monviewer,(PetscErrorCode (*)(void**))PetscViewerDestroy);
107:     }
108:     flg = PETSC_FALSE;
109:     PetscOptionsBool("-mfn_monitor_lg","Monitor error estimate graphically","MFNMonitorSet",flg,&flg,NULL);
110:     if (flg) {
111:       MFNMonitorSet(mfn,MFNMonitorLG,NULL,NULL);
112:     }
113:   /* -----------------------------------------------------------------------*/

115:     PetscOptionsName("-mfn_view","Print detailed information on solver used","MFNView",0);

117:     if (mfn->ops->setfromoptions) {
118:       (*mfn->ops->setfromoptions)(mfn);
119:     }
120:     PetscObjectProcessOptionsHandlers((PetscObject)mfn);
121:   PetscOptionsEnd();

123:   if (!mfn->V) { MFNGetBV(mfn,&mfn->V); }
124:   BVSetFromOptions(mfn->V);
125:   if (!mfn->ds) { MFNGetDS(mfn,&mfn->ds); }
126:   DSSetFromOptions(mfn->ds);
127:   PetscRandomSetFromOptions(mfn->rand);
128:   return(0);
129: }

133: /*@
134:    MFNGetTolerances - Gets the tolerance and maximum iteration count used
135:    by the MFN convergence tests.

137:    Not Collective

139:    Input Parameter:
140: .  mfn - the matrix function context

142:    Output Parameters:
143: +  tol - the convergence tolerance
144: -  maxits - maximum number of iterations

146:    Notes:
147:    The user can specify NULL for any parameter that is not needed.

149:    Level: intermediate

151: .seealso: MFNSetTolerances()
152: @*/
153: PetscErrorCode MFNGetTolerances(MFN mfn,PetscReal *tol,PetscInt *maxits)
154: {
157:   if (tol)    *tol    = mfn->tol;
158:   if (maxits) *maxits = mfn->max_it;
159:   return(0);
160: }

164: /*@
165:    MFNSetTolerances - Sets the tolerance and maximum iteration count used
166:    by the MFN convergence tests.

168:    Logically Collective on MFN

170:    Input Parameters:
171: +  mfn - the matrix function context
172: .  tol - the convergence tolerance
173: -  maxits - maximum number of iterations to use

175:    Options Database Keys:
176: +  -mfn_tol <tol> - Sets the convergence tolerance
177: -  -mfn_max_it <maxits> - Sets the maximum number of iterations allowed

179:    Notes:
180:    Use PETSC_DEFAULT for either argument to assign a reasonably good value.

182:    Level: intermediate

184: .seealso: MFNGetTolerances()
185: @*/
186: PetscErrorCode MFNSetTolerances(MFN mfn,PetscReal tol,PetscInt maxits)
187: {
192:   if (tol == PETSC_DEFAULT) {
193:     mfn->tol = PETSC_DEFAULT;
194:     mfn->setupcalled = 0;
195:   } else {
196:     if (tol <= 0.0) SETERRQ(PetscObjectComm((PetscObject)mfn),PETSC_ERR_ARG_OUTOFRANGE,"Illegal value of tol. Must be > 0");
197:     mfn->tol = tol;
198:   }
199:   if (maxits == PETSC_DEFAULT || maxits == PETSC_DECIDE) {
200:     mfn->max_it = 0;
201:     mfn->setupcalled = 0;
202:   } else {
203:     if (maxits <= 0) SETERRQ(PetscObjectComm((PetscObject)mfn),PETSC_ERR_ARG_OUTOFRANGE,"Illegal value of maxits. Must be > 0");
204:     mfn->max_it = maxits;
205:   }
206:   return(0);
207: }

237: /*@
238:    MFNSetDimensions - Sets the dimension of the subspace to be used by the solver.

240:    Logically Collective on MFN

242:    Input Parameters:
243: +  mfn - the matrix function context
244: -  ncv - the maximum dimension of the subspace to be used by the solver

246:    Options Database Keys:
247: .  -mfn_ncv <ncv> - Sets the dimension of the subspace

249:    Notes:
250:    Use PETSC_DEFAULT for ncv to assign a reasonably good value, which is
251:    dependent on the solution method.

253:    Level: intermediate

255: .seealso: MFNGetDimensions()
256: @*/
257: PetscErrorCode MFNSetDimensions(MFN mfn,PetscInt ncv)
258: {
262:   if (ncv == PETSC_DECIDE || ncv == PETSC_DEFAULT) {
263:     mfn->ncv = 0;
264:   } else {
265:     if (ncv<1) SETERRQ(PetscObjectComm((PetscObject)mfn),PETSC_ERR_ARG_OUTOFRANGE,"Illegal value of ncv. Must be > 0");
266:     mfn->ncv = ncv;
267:   }
268:   mfn->setupcalled = 0;
269:   return(0);
270: }

274: /*@
275:    MFNSetFunction - Specifies the function to be computed.

277:    Logically Collective on MFN

279:    Input Parameters:
280: +  mfn      - the matrix function context
281: -  fun      - a known function

283:    Options Database Keys:
284: .  -mfn_exp - matrix exponential

286:    Level: beginner

288: .seealso: MFNSetOperator(), MFNSetType(), MFNGetFunction(), SlepcFunction
289: @*/
290: PetscErrorCode MFNSetFunction(MFN mfn,SlepcFunction fun)
291: {
295:   switch (fun) {
296:     case SLEPC_FUNCTION_EXP:
297:       break;
298:     default:
299:       SETERRQ(PetscObjectComm((PetscObject)mfn),PETSC_ERR_ARG_WRONG,"Unknown function");
300:   }
301:   mfn->function = fun;
302:   return(0);
303: }

334:    MFNSetScaleFactor - Sets the scale factor to multiply the matrix (the

453: /*@C
454:    MFNSetOptionsPrefix - Sets the prefix used for searching for all
455:    MFN options in the database.

457:    Logically Collective on MFN

459:    Input Parameters:
460: +  mfn - the matrix function context
461: -  prefix - the prefix string to prepend to all MFN option requests

463:    Notes:
464:    A hyphen (-) must NOT be given at the beginning of the prefix name.
465:    The first character of all runtime options is AUTOMATICALLY the
466:    hyphen.

468:    For example, to distinguish between the runtime options for two
469:    different MFN contexts, one could call
470: .vb
471:       MFNSetOptionsPrefix(mfn1,"fun1_")
472:       MFNSetOptionsPrefix(mfn2,"fun2_")
473: .ve

475:    Level: advanced

477: .seealso: MFNAppendOptionsPrefix(), MFNGetOptionsPrefix()
478: @*/
479: PetscErrorCode MFNSetOptionsPrefix(MFN mfn,const char *prefix)
480: {

485:   if (!mfn->V) { MFNGetBV(mfn,&mfn->V); }
486:   BVSetOptionsPrefix(mfn->V,prefix);
487:   if (!mfn->ds) { MFNGetDS(mfn,&mfn->ds); }
488:   DSSetOptionsPrefix(mfn->ds,prefix);
489:   PetscObjectSetOptionsPrefix((PetscObject)mfn,prefix);
490:   return(0);
491: }

495: /*@C
496:    MFNAppendOptionsPrefix - Appends to the prefix used for searching for all
497:    MFN options in the database.

499:    Logically Collective on MFN

501:    Input Parameters:
502: +  mfn - the matrix function context
503: -  prefix - the prefix string to prepend to all MFN option requests

505:    Notes:
506:    A hyphen (-) must NOT be given at the beginning of the prefix name.
507:    The first character of all runtime options is AUTOMATICALLY the hyphen.

509:    Level: advanced

511: .seealso: MFNSetOptionsPrefix(), MFNGetOptionsPrefix()
512: @*/
513: PetscErrorCode MFNAppendOptionsPrefix(MFN mfn,const char *prefix)
514: {

519:   if (!mfn->V) { MFNGetBV(mfn,&mfn->V); }
520:   BVSetOptionsPrefix(mfn->V,prefix);
521:   if (!mfn->ds) { MFNGetDS(mfn,&mfn->ds); }
522:   DSSetOptionsPrefix(mfn->ds,prefix);
523:   PetscObjectAppendOptionsPrefix((PetscObject)mfn,prefix);
524:   return(0);
525: }

529: /*@C
530:    MFNGetOptionsPrefix - Gets the prefix used for searching for all
531:    MFN options in the database.

533:    Not Collective

535:    Input Parameters:
536: .  mfn - the matrix function context

538:    Output Parameters:
539: .  prefix - pointer to the prefix string used is returned

541:    Notes: On the fortran side, the user should pass in a string 'prefix' of
542:    sufficient length to hold the prefix.

544:    Level: advanced

546: .seealso: MFNSetOptionsPrefix(), MFNAppendOptionsPrefix()
547: @*/
548: PetscErrorCode MFNGetOptionsPrefix(MFN mfn,const char *prefix[])
549: {

555:   PetscObjectGetOptionsPrefix((PetscObject)mfn,prefix);
556:   return(0);
557: }
