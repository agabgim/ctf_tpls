 70: static PetscErrorCode SVDOneSideTRLanczosMGS(SVD svd,PetscReal *alpha,PetscReal *beta,BV V,BV U,PetscInt nconv,PetscInt l,PetscInt n,PetscScalar* work)
 71: {
 73:   PetscReal      a,b;
 74:   PetscInt       i,k=nconv+l;
 75:   Vec            ui,ui1,vi;

 78:   BVGetColumn(V,k,&vi);
 79:   BVGetColumn(U,k,&ui);
 80:   SVDMatMult(svd,PETSC_FALSE,vi,ui);
 81:   BVRestoreColumn(V,k,&vi);
 82:   BVRestoreColumn(U,k,&ui);
 83:   if (l>0) {
 84:     for (i=0;i<l;i++) work[i]=beta[i+nconv];
 85:     BVMultColumn(U,-1.0,1.0,k,work);
 86:   }
 87:   BVNormColumn(U,k,NORM_2,&a);
 88:   BVScaleColumn(U,k,1.0/a);
 89:   alpha[k] = a;

 91:   for (i=k+1;i<n;i++) {
 92:     BVGetColumn(V,i,&vi);
 93:     BVGetColumn(U,i-1,&ui1);
 94:     SVDMatMult(svd,PETSC_TRUE,ui1,vi);
 95:     BVRestoreColumn(V,i,&vi);
 96:     BVRestoreColumn(U,i-1,&ui1);
 97:     BVOrthogonalizeColumn(V,i,NULL,&b,NULL);
 98:     BVScaleColumn(V,i,1.0/b);
 99:     beta[i-1] = b;

101:     BVGetColumn(V,i,&vi);
102:     BVGetColumn(U,i,&ui);
103:     SVDMatMult(svd,PETSC_FALSE,vi,ui);
104:     BVRestoreColumn(V,i,&vi);
105:     BVGetColumn(U,i-1,&ui1);
106:     VecAXPY(ui,-b,ui1);
107:     BVRestoreColumn(U,i-1,&ui1);
108:     BVRestoreColumn(U,i,&ui);
109:     BVNormColumn(U,i,NORM_2,&a);
110:     BVScaleColumn(U,i,1.0/a);
111:     alpha[i] = a;
112:   }

114:   BVGetColumn(V,n,&vi);
115:   BVGetColumn(U,n-1,&ui1);
116:   SVDMatMult(svd,PETSC_TRUE,ui1,vi);
117:   BVRestoreColumn(V,n,&vi);
118:   BVRestoreColumn(U,n-1,&ui1);
119:   BVOrthogonalizeColumn(V,n,NULL,&b,NULL);
120:   beta[n-1] = b;
121:   return(0);
122: }

126: /*
127:   Custom CGS orthogonalization, preprocess after first orthogonalization
128: */
129: static PetscErrorCode SVDOrthogonalizeCGS(BV V,PetscInt i,PetscScalar* h,PetscReal a,BVOrthogRefineType refine,PetscReal eta,PetscReal *norm)
130: {
132:   PetscReal      sum,onorm;
133:   PetscScalar    dot;
134:   PetscInt       j;

137:   switch (refine) {
138:   case BV_ORTHOG_REFINE_NEVER:
139:     BVNormColumn(V,i,NORM_2,norm);
140:     break;
141:   case BV_ORTHOG_REFINE_ALWAYS:
142:     BVSetActiveColumns(V,0,i);
143:     BVDotColumn(V,i,h);
144:     BVMultColumn(V,-1.0,1.0,i,h);
145:     BVNormColumn(V,i,NORM_2,norm);
146:     break;
147:   case BV_ORTHOG_REFINE_IFNEEDED:
148:     dot = h[i];
149:     onorm = PetscSqrtReal(PetscRealPart(dot)) / a;
150:     sum = 0.0;
151:     for (j=0;j<i;j++) {
152:       sum += PetscRealPart(h[j] * PetscConj(h[j]));
153:     }
154:     *norm = PetscRealPart(dot)/(a*a) - sum;
155:     if (*norm>0.0) *norm = PetscSqrtReal(*norm);
156:     else {
157:       BVNormColumn(V,i,NORM_2,norm);
158:     }
159:     if (*norm < eta*onorm) {
160:       BVSetActiveColumns(V,0,i);
161:       BVDotColumn(V,i,h);
162:       BVMultColumn(V,-1.0,1.0,i,h);
163:       BVNormColumn(V,i,NORM_2,norm);
164:     }
165:     break;
166:   }
167:   return(0);
168: }

172: static PetscErrorCode SVDOneSideTRLanczosCGS(SVD svd,PetscReal *alpha,PetscReal *beta,BV V,BV U,PetscInt nconv,PetscInt l,PetscInt n,PetscScalar* work)
173: {
174:   PetscErrorCode     ierr;
175:   PetscReal          a,b,eta;
176:   PetscInt           i,j,k=nconv+l;
177:   Vec                ui,ui1,vi;
178:   BVOrthogRefineType refine;

181:   BVGetColumn(V,k,&vi);
182:   BVGetColumn(U,k,&ui);
183:   SVDMatMult(svd,PETSC_FALSE,vi,ui);
184:   BVRestoreColumn(V,k,&vi);
185:   BVRestoreColumn(U,k,&ui);
186:   if (l>0) {
187:     for (i=0;i<l;i++) work[i]=beta[i+nconv];
188:     BVMultColumn(U,-1.0,1.0,k,work);
189:   }
190:   BVGetOrthogonalization(V,NULL,&refine,&eta);

192:   for (i=k+1;i<n;i++) {
193:     BVGetColumn(V,i,&vi);
194:     BVGetColumn(U,i-1,&ui1);
195:     SVDMatMult(svd,PETSC_TRUE,ui1,vi);
196:     BVRestoreColumn(V,i,&vi);
197:     BVRestoreColumn(U,i-1,&ui1);
198:     BVNormColumn(U,i-1,NORM_2,&a);
199:     if (refine == BV_ORTHOG_REFINE_IFNEEDED) {
200:       BVSetActiveColumns(V,0,i+1);
201:       BVGetColumn(V,i,&vi);
202:       BVDotVec(V,vi,work);
203:       BVRestoreColumn(V,i,&vi);
204:       BVSetActiveColumns(V,0,i);
205:     } else {
206:       BVSetActiveColumns(V,0,i);
207:       BVDotColumn(V,i,work);
208:     }

210:     BVScaleColumn(U,i-1,1.0/a);
211:     for (j=0;j<i;j++) work[j] = work[j] / a;
212:     BVMultColumn(V,-1.0,1.0/a,i,work);
213:     SVDOrthogonalizeCGS(V,i,work,a,refine,eta,&b);
214:     BVScaleColumn(V,i,1.0/b);

216:     BVGetColumn(V,i,&vi);
217:     BVGetColumn(U,i,&ui);
218:     BVGetColumn(U,i-1,&ui1);
219:     SVDMatMult(svd,PETSC_FALSE,vi,ui);
220:     VecAXPY(ui,-b,ui1);
221:     BVRestoreColumn(V,i,&vi);
222:     BVRestoreColumn(U,i,&ui);
223:     BVRestoreColumn(U,i-1,&ui1);

225:     alpha[i-1] = a;
226:     beta[i-1] = b;
227:   }

229:   BVGetColumn(V,n,&vi);
230:   BVGetColumn(U,n-1,&ui1);
231:   SVDMatMult(svd,PETSC_TRUE,ui1,vi);
232:   BVRestoreColumn(V,n,&vi);
233:   BVRestoreColumn(U,n-1,&ui1);

235:   BVNormColumn(svd->U,n-1,NORM_2,&a);
236:   if (refine == BV_ORTHOG_REFINE_IFNEEDED) {
237:     BVSetActiveColumns(V,0,n+1);
238:     BVGetColumn(V,n,&vi);
239:     BVDotVec(V,vi,work);
240:     BVRestoreColumn(V,n,&vi);
241:   } else {
242:     BVSetActiveColumns(V,0,n);
243:     BVDotColumn(V,n,work);
244:   }

246:   BVScaleColumn(U,n-1,1.0/a);
247:   for (j=0;j<n;j++) work[j] = work[j] / a;
248:   BVMultColumn(V,-1.0,1.0/a,n,work);
249:   SVDOrthogonalizeCGS(V,n,work,a,refine,eta,&b);
250:   BVSetActiveColumns(V,nconv,n);
251:   alpha[n-1] = a;
252:   beta[n-1] = b;
253:   return(0);
254: }

258: PetscErrorCode SVDSolve_TRLanczos(SVD svd)
259: {
261:   SVD_TRLANCZOS  *lanczos = (SVD_TRLANCZOS*)svd->data;
262:   PetscReal      *alpha,*beta,lastbeta,norm;
263:   PetscScalar    *Q,*swork=NULL,*w;
264:   PetscInt       i,k,l,nv,ld;
265:   Mat            U,VT;
266:   PetscBool      conv;
267:   BVOrthogType   orthog;

270:   /* allocate working space */
271:   DSGetLeadingDimension(svd->ds,&ld);
272:   BVGetOrthogonalization(svd->V,&orthog,NULL,NULL);
273:   PetscMalloc1(ld,&w);
274:   if (lanczos->oneside) {
275:     PetscMalloc1(svd->ncv+1,&swork);
276:   }

278:   /* normalize start vector */
279:   if (!svd->nini) {
280:     BVSetRandomColumn(svd->V,0,svd->rand);
281:     BVNormColumn(svd->V,0,NORM_2,&norm);
282:     BVScaleColumn(svd->V,0,1.0/norm);
283:   }

285:   l = 0;
286:   while (svd->reason == SVD_CONVERGED_ITERATING) {
287:     svd->its++;

289:     /* inner loop */
290:     nv = PetscMin(svd->nconv+svd->mpd,svd->ncv);
291:     BVSetActiveColumns(svd->V,svd->nconv,nv);
292:     BVSetActiveColumns(svd->U,svd->nconv,nv);
293:     DSGetArrayReal(svd->ds,DS_MAT_T,&alpha);
294:     beta = alpha + ld;
295:     if (lanczos->oneside) {
296:       if (orthog == BV_ORTHOG_MGS) {
297:         SVDOneSideTRLanczosMGS(svd,alpha,beta,svd->V,svd->U,svd->nconv,l,nv,swork);
298:       } else {
299:         SVDOneSideTRLanczosCGS(svd,alpha,beta,svd->V,svd->U,svd->nconv,l,nv,swork);
300:       }
301:     } else {
302:       SVDTwoSideLanczos(svd,alpha,beta,svd->V,svd->U,svd->nconv+l,nv);
303:     }
304:     lastbeta = beta[nv-1];
305:     DSRestoreArrayReal(svd->ds,DS_MAT_T,&alpha);
306:     BVScaleColumn(svd->V,nv,1.0/lastbeta);

308:     /* compute SVD of general matrix */
309:     DSSetDimensions(svd->ds,nv,nv,svd->nconv,svd->nconv+l);
310:     if (l==0) {
311:       DSSetState(svd->ds,DS_STATE_INTERMEDIATE);
312:     } else {
313:       DSSetState(svd->ds,DS_STATE_RAW);
314:     }
315:     DSSolve(svd->ds,w,NULL);
316:     DSSort(svd->ds,w,NULL,NULL,NULL,NULL);

318:     /* compute error estimates */
319:     k = 0;
320:     conv = PETSC_TRUE;
321:     DSGetArray(svd->ds,DS_MAT_U,&Q);
322:     DSGetArrayReal(svd->ds,DS_MAT_T,&alpha);
323:     beta = alpha + ld;
324:     for (i=svd->nconv;i<nv;i++) {
325:       svd->sigma[i] = PetscRealPart(w[i]);
326:       beta[i] = PetscRealPart(Q[nv-1+i*ld])*lastbeta;
327:       svd->errest[i] = PetscAbsScalar(beta[i]);
328:       if (svd->sigma[i] > svd->tol) svd->errest[i] /= svd->sigma[i];
329:       if (conv) {
330:         if (svd->errest[i] < svd->tol) k++;
331:         else conv = PETSC_FALSE;
332:       }
333:     }
334:     DSRestoreArrayReal(svd->ds,DS_MAT_T,&alpha);
335:     DSRestoreArray(svd->ds,DS_MAT_U,&Q);

337:     /* check convergence and update l */
338:     if (svd->its >= svd->max_it) svd->reason = SVD_DIVERGED_ITS;
339:     if (svd->nconv+k >= svd->nsv) svd->reason = SVD_CONVERGED_TOL;
340:     if (svd->reason != SVD_CONVERGED_ITERATING) l = 0;
341:     else l = PetscMax((nv-svd->nconv-k)/2,0);

343:     /* compute converged singular vectors and restart vectors */
344:     DSGetMat(svd->ds,DS_MAT_VT,&VT);
345:     BVMultInPlaceTranspose(svd->V,VT,svd->nconv,svd->nconv+k+l);
346:     MatDestroy(&VT);
347:     DSGetMat(svd->ds,DS_MAT_U,&U);
348:     BVMultInPlace(svd->U,U,svd->nconv,svd->nconv+k+l);
349:     MatDestroy(&U);

351:     /* copy the last vector to be the next initial vector */
352:     if (svd->reason == SVD_CONVERGED_ITERATING) {
353:       BVCopyColumn(svd->V,nv,svd->nconv+k+l);
354:     }

356:     svd->nconv += k;
357:     SVDMonitor(svd,svd->its,svd->nconv,svd->sigma,svd->errest,nv);
358:   }

360:   /* orthonormalize U columns in one side method */
361:   if (lanczos->oneside) {
362:     for (i=0;i<svd->nconv;i++) {
363:       BVOrthogonalizeColumn(svd->U,i,NULL,&norm,NULL);
364:       BVScaleColumn(svd->U,i,1.0/norm);
365:     }
366:   }

368:   /* free working space */
369:   PetscFree(w);
370:   if (swork) { PetscFree(swork); }
371:   return(0);
372: }

405: /*@
406:    SVDTRLanczosSetOneSide - Indicate if the variant of the Lanczos method
407:    to be used is one-sided or two-sided.

409:    Logically Collective on SVD

411:    Input Parameters:
412: +  svd     - singular value solver
413: -  oneside - boolean flag indicating if the method is one-sided or not

415:    Options Database Key:
416: .  -svd_trlanczos_oneside <boolean> - Indicates the boolean flag

418:    Note:
419:    By default, a two-sided variant is selected, which is sometimes slightly
420:    more robust. However, the one-sided variant is faster because it avoids
421:    the orthogonalization associated to left singular vectors.

423:    Level: advanced

425: .seealso: SVDLanczosSetOneSide()
426: @*/
427: PetscErrorCode SVDTRLanczosSetOneSide(SVD svd,PetscBool oneside)
428: {

434:   PetscTryMethod(svd,"SVDTRLanczosSetOneSide_C",(SVD,PetscBool),(svd,oneside));
435:   return(0);
436: }

449:    Output Parameters:
